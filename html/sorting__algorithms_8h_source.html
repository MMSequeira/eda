1 // `sorting_algorithms.h` &ndash; Rotinas de ordenação de _arrays_ de `double`
2 // ===========================================================================
3 //
4 // Este é o ficheiro de cabeçalho ou de interface correspondente ao ficheiro de
5 // implementação [`sorting_algorithms.c`](sorting_algorithms.c.html). Ambos
6 // correspondem ao módulo físico `sorting_algorithms`, cujo objectivo é fornecer
7 // um conjunto de ferramentas para lidar com a ordenação de _arrays_ de
8 // `double`. As rotinas fornecidas implementam alguns dos mais conhecidos
9 // algoritmos de ordenação:
10 //
11 // - Ordenação por bolha ou _bubble sort_.
12 // - Ordenação por selecção ou _selection sort_.
13 // - Ordenação por inserção ou _insertion sort_.
14 // - Ordenação de Shell ou _Shell sort_.
15 // - Ordenação rápida ou _quicksort_.
16 // - Ordenação por fusão ou _merge sort_.
17 //
18 // Este módulo foi concebido para o estudo da algoritmia. Por isso, para além de
19 // uma implementação «normal» de cada um dos algoritmos, existe uma outra que é
20 // em tudo igual mas que regista o número de operações elementares realizadas.
21 //
22 // Note que optámos por _não_ incluir comentários de documentação
23 // [Doxygen](http://doxygen.org/) em nenhum dos módulos deste programa.
24 
25 // A usual protecção contra os efeitos nefastos da inclusão múltipla.
26 #ifndef ISLA_EDA_SORTING_ALGORITHMS_H_INCLUDED

36 // As rotinas de ordenação que registam o número de operações elementares
37 // realizadas durante a ordenação de uma _array_ recorrem a esta estrutura para
38 // guardar as suas contagens.
39 struct algorithm_counts {
40  // Este campo regista o número de comparações envolvendo pelo menos um
41  // item do _array_ em ordenação.
42  int comparisons;
43  // Este campo regista o número de trocas de valores entre pares itens do
44  // _array_ em ordenação.
45  int swaps;
46  // Este campo regista o número de cópias de valores envolvendo pelo
47  // menos um item do _array_ em ordenação.
48  int copies;
49 };
50 
51 // Esta estrutura guarda informação acerca de um algoritmo.
52 struct sorting_algorithm {
53  // Este campo guarda o nome do algoritmo na forma de uma cadeia de
54  // caracteres.
55  const char* name;
56  // Este campo guarda um ponteiro para a rotina que implementa o
57  // algoritmo em causa e que _não_ efectua contagem de operações
58  // elementares.
59  bool (*sort)(int, double[]);
60  // Este campo guarda um ponteiro para a rotina que implementa o
61  // algoritmo em causa e que efectua contagem de operações elementares.
62  bool (*sort_and_count)(int, double[], struct algorithm_counts*);
63 };
