42 // ### Definição da estrutura `struct sequence_of_long`
43 //
44 // Esta estrutura contém três campos ou atributos:
45 //
46 // - `terms` &ndash; Um ponteiro para o _array_ dinâmico que contém os termos da
47 // sucessão.
48 //
49 // - `length` &ndash; Inteiro guardando o comprimento actual da sucessão, i.e.,
50 // o seu número de termos. Note que o comprimento da sucessão é sempre
51 // inferior ou igual ao comprimento do `array` que guarda os termos, a que
52 // chamamos «capacidade».
53 //
54 // - `capacity` &ndash; Inteiro guardando o comprimento actual do `array`
55 // dinâmico que guarda os `length` termos actualmente na sucessão.
56 struct sequence_of_longs {
57  long *terms;
58  int length;
59  int capacity;
60 };
61 
62 // ### Implementação dos procedimentos que imprimem as sucessões
63 //
64 // Note-se que todas as rotinas de manipulação das sucessões, com excepção
65 // naturalmente do construtor, recebem um ponteiro para a sucessão em causa como
66 // primeiro argumento. Convencionou-se chamar `sl` ao correspondente parâmetro
67 // em todas as definições de rotinas abaixo, com a excepção referida.
68 void SEQL_print(struct sequence_of_longs *sl)
69 {
70  // O procedimento `putchar()` imprime no ecrã o caractere que recebe
71  // como argumento. Neste caso trata-se da chaveta inicial da
72  // representação textual da sucessão.
73  putchar('{');
74 
75  // Depois de impressa a chaveta inicial, imprimimos cada um dos termos
76  // da sucessão usando um ciclo `for`. Note que o ciclo imprime apenas os
77  // termos da sucessão, e não todos os itens do _array_ dinâmico que os
78  // armazena. Daí que a sua guarda se refira a `sl->length`, e não a
79  // `sl->capacity`.
80  for (int i = 0; i != sl->length; i++) {
81  // Os termos são separados uns dos outros por `,␣` (usamos o
82  // caractere `␣` para representar o espaço). Assim, podemos
83  // preceder cada termo do separador _com excepção do primeiro_.
84  if (i != 0)
85  printf(", ");
86 
87  // Imprimimos cada termo usando a especificação de conversão
88  // `%ld`, e não simplesmente `%d`, pois os termos da sucessão
89  // são `long`.
90  printf("%ld", sl->terms[i]);
91  }
92 
93  // Terminamos imprimindo a chaveta de fecho.
94  putchar('}');
95 }

107 // ### Implementação do construtor do TAD
108 //
109 struct sequence_of_longs *SEQL_new(void)
110 {
111  // Reservamos primeiro espaço para uma nova `struct sequence_of_longs`
112  // usando a rotina `malloc()`. Obtemos a quantidade de unidades de
113  // memória a reservar usando o operador `sizeof`. Guardamos o endereço
114  // dessa nova variável dinâmica no ponteiro `sl`, através do qual se
115  // passará a manipular a nova sucessão de `long`. No final do construtor
116  // este ponteiro será devolvido.
117  struct sequence_of_longs *sl = malloc(sizeof(struct sequence_of_longs));
118 
119  // De seguida inicializamos os campos ou atributos da rotina com valores
120  // apropriados para uma sucessão vazia. O comprimento de uma sucessão
121  // vazia é naturalmente 0. A capacidade do _array_ dinâmico que guarda
122  // os termos também poderia ser inicialmente 0. No entanto, como se verá
123  // mais abaixo, a estratégia usada é a de duplicar a capacidade sempre
124  // que ela está esgotada e se pretende adicionar um termo à sucessão.
125  // Ora, uma capacidade de 0 estaria esgotada logo na primeira adição de
126  // um termo e, pior, o dobro de 0 é... 0! É mais sensato, por isso,
127  // começar com um capacidade maior que zero. Começar com 1 é uma
128  // possibilidade, embora qualquer outro valor positivo fosse igualmente
129  // eficaz. Quanto maior for a capacidade inicial, maior a eficiência das
130  // adições, mas pior a eficiência na utilização da memória. Daí que a
131  // escolha de uma capacidade inicial unitária não seja totalmente
132  // desprovida de sentido.
133  //
134  // Ah! E não se esqueça que o compilador traduz `sl->length` por
135  // `(*sl).length`. Ou seja, o significado de `sl->length` é «o campo
136  // `length` da estrutura apontada pelo ponteiro `sl`».
137  sl->length = 0;
138  sl->capacity = 1;
139 
140  // Inicializados os campos que guardam o comprimento da sucessão e a
141  // capacidade do _array_ dinâmico que guarda os termos da sucessão, há
142  // que reservar espaço de memória para este _array_ e guardar o endereço
143  // do seu primeiro item no campo `terms`. É o que se faz aqui,
144  // recorrendo à rotina `malloc()`. Note-se que se reserva espaço
145  // suficiente para `sl->capacity` termos.
146  sl->terms = malloc(sl->capacity * sizeof(long));
147 
148  // Terminada a construção da nova sucessão, há que devolver o seu
149  // endereço (ou ponteiro), guardado na variável `sl`. Será através dele
150  // que o código cliente fará uso da sucessão, passando-o como primeiro
151  // argumento às rotinas fornecidas por este módulo físico.
152  return sl;
153 }

167 // ### Implementação do _modificador_ de adição de um novo termo à sucessão
168 //
169 // Dá-se o nome de modificador a um procedimento que altera uma instância de um
170 // TAD. Neste caso o modificador é o procedimento `SEQL_add()`, que altera a
171 // sucessão cujo ponteiro é `sl` adicionando-lhe um novo termo `new_term`.
172 //
173 void SEQL_add(struct sequence_of_longs *sl, long new_term)
174 {
175  // Antes de adicionar o termo há que verificar se o _array_ dinâmico
176  // onde será guardado tem capacidade suficiente para ele. Se o
177  // comprimento actual da sucessão for igual à capacidade desse _array_,
178  // então a capacidade está esgotada, sendo necessário aumentá-la.
179  if (sl->length == sl->capacity)
180  // Para aumentar a capacidade, duplicamos o valor do campo
181  // `sl->capacity` e usamos a rotina `realloc()` para reservar um
182  // novo _array_ dinâmico com essa nova capacidade _e incluindo
183  // todos os termos que já constam na sucessão_. É este último
184  // requisito que nos leva a usar a rotina `realloc()`, e não a
185  // rotina `malloc()`. A rotina `realloc()`, se precisar de
186  // reservar nova memória, i.e., se não conseguir simplesmente
187  // estender o _array_ existente, copiará automaticamente os
188  // termos da memória original. É por isso que o primeiro
189  // argumento passado a `realloc()` é um ponteiro para o _array_
190  // original.
191  sl->terms = realloc(sl->terms,
192  (sl->capacity *= 2) * sizeof(long));
193 
194  // Estas linhas poderiam ser substituídas pelo código abaixo. À parte a
195  // possibilidade de estender o _array_ original, de que a rotina
196  // `realloc()` pode tirar partido, o resultado seria exactamente o
197  // mesmo. Preferimos, naturalmente, a versão mais curta... Quanto à
198  // duplicação da capacidade numa instrução à parte, é certamente
199  // preferível, mas é tão comum (idiomático) encontrar o código
200  // condensado numa única instrução, que optámos pela versão mais
201  // críptica.
202  //
203  // ```C
204  // if (sl->length == sl->capacity) {
205  // sl->capacity *= 2;
206  // long *new_array = malloc(sl->capacity * sizeof(long));
207  // for (int i = 0; i != sl->length; i++)
208  // new_array[i] = sl->terms[i];
209  // free(sl->terms);
210  // sl->terms = new_array;
211  // }
212  // ```
213 
214  // Finalmente, guardamos o novo termo `new_term` no local apropriado do
215  // _array_, que neste ponto já tem certamente (hmmmm... será?)
216  // capacidade suficiente, i.e., na posição `sl->length`. Depois,
217  // incrementamos o comprimento da sucessão. Mais uma vez, a condensação
218  // de duas alterações numa única instrução é uma má prática, mas tão
219  // generalizada na programação em C que já se tornou idiomática. A
220  // alternativa, mais clara, seria usar:
221  //
222  // ```C
223  // sl->terms[sl->length] = new_term;
224  // sl->length++;
225  // ```
226  sl->terms[sl->length++] = new_term;
227 }

229 // ### Implementação do _inspector_ de termo
230 //
231 // Este inspector permite obter o termo da sucessão na posição dada por _index_.

233 long SEQL_term(struct sequence_of_longs *sl, int index)
234 {
235  // Note a utilização da indexação do item `index` do _array_ dinâmico
236  // `sl->terms`, ao qual se acede usando o operador `->` sobre o ponteiro
237  // `sl` para a estrutura que representa a sucessão.
238  return sl->terms[index];
239 }
