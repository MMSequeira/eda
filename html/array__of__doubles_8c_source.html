60 double *read_double_array_from(const char *const file_name, int *const length)
61 {
62  assert(file_name != NULL);
63  assert(length != NULL);
64 
65  // Começamos por abrir um novo canal de leitura para o ficheiro cujo
66  // nome recebemos como argumento.
67  FILE *file = fopen(file_name, "r");
68 
69  // Em C, o código que lida com casos de erro e casos excepcionais
70  // emaranha-se com o código «normal». Isso torna todo o código menos
71  // legível. Para evitar este problema a melhor solução é mesmo mudar de
72  // linguagem. :-)
73  if (file == NULL)
74  return NULL;
75 
76  // O _array_ onde iremos guardando os itens lidos tem de ir crescendo à
77  // medida que a leitura prossegue. Em cada instante, o _array_ terá como
78  // capacidade o valor guardado nesta variável. O número de itens já
79  // lidos e guardados no _array_ será sempre inferior ou igual à sua
80  // capacidade.
81  int capacity = 32;
82 
83  // Criamos o _array_ com a capacidade definida inicialmente.
84  double *items = new_double_array_of(capacity);
85 
86  // No caso de não se conseguir criar o novo _array_ dinâmico, fechamos o
87  // canal aberto para o ficheiro em leitura e retornamos devolvendo o
88  // valor especial `NULL`, de modo a assinalar o erro.
89  if (items == NULL) {
90  fclose(file);
91  return NULL;
92  }
93 
94  // O ciclo de leitura é feito recorrendo a duas variáveis locais. A
95  // primeira, `i`, indica em cada momento, e simultaneamente, quantos
96  // itens já foram lidos e o índice do item do _array_ onde será guardado
97  // o próximo valor lido do ficheiro. A segunda variável é usada para ler
98  // cada um dos itens. Em caso de sucesso na leitura, o valor contido
99  // neste variável será atribuído ao item apropriado do _array_.
100  int i = 0;
101  double item;
102  // A guarda do ciclo inclui a própria operação de leitura. Esta forma de
103  // escrever o ciclo é idiomática do C e seria considerada uma má prática
104  // em qualquer outra linguagem. Fazemos a leitura usando a usual rotina
105  // `fscanf()`, recorrendo à especificação de conversão `%lg`, onde a
106  // utilização de `l` é necessária por se tratar de `doubles`. A leitura
107  // é realizada para a variável `item`, pelo que passamos o seu endereço
108  // à rotina `fscanf()`. Se a leitura tiver sucesso, a rotina devolve o
109  // valor 1, pois terá sido atribuído exactamente um valor, nomeadamente
110  // o valor lido do ficheiro e guardado na variável `item`.
111  while (fscanf(file, "%lg", &item) == 1) {
112  // Se o número de itens lidos com sucesso e guardados no _array_
113  // for igual à capacidade do _array_, então esta encontra-se
114  // esgotada. Sendo necessário aumentar a capacidade do _array_,
115  // pode-se demonstrar que uma forma eficiente de o fazer é
116  // duplicá-la, pois leva a uma número médio de cópias por item
117  // do _array_ que é constante e pequeno. Note que o valor da
118  // variável `capacidade` é aumentado na mesma instrução em que
119  // se aumenta a capacidade do _array_, pelo que essa instrução
120  // terá dois efeitos distintos. Mais uma vez, trata-se de código
121  // muito discutível, mas idiomático na linguagem C.
122  if (i == capacity)
123  items = resize_double_array_to(items, capacity *= 2);
124  // Atribuímos o valor lido ao item do _array_ com índice `i` _e
125  // incrementamos o próprio `i`_! O valor de `i` usado na
126  // indexação é o valor original de `i`, antes da incrementação.
127  // Trata-se, outra vez, de código muito discutível, mas
128  // idiomático na linguagem C.
129  items[i++] = item;
130  }
131 
132  // Finda a leitura, a variável `i` contém o número de itens lidos e
133  // guardados no _array_. Guardamos o seu valor na variável apontada por
134  // `length`.
135  *length = i;
136 
137  // Se necessário, ajustamos a capacidade do _array_ de modo a
138  // corresponder exactamente ao número de itens lidos, de modo a garantir
139  // que não há desperdício de memória.
140  if (*length != capacity)
141  items = resize_double_array_to(items, *length);
142 
143  // Fechamos o canal de entrada.
144  fclose(file);
145 
146  // Retornamos, devolvendo o endereço do primeiro item do _array_
147  // dinâmico criado, que contém os valores lidos do ficheiro.
148  return items;
149 }

162 bool double_arrays_equal(int length,
163  const double first[length], const double second[length])
164 {
165  assert(length >= 0);
166  assert(length == 0 || first != NULL);
167  assert(length == 0 || second != NULL);
168 
169  // Note a utilização de um ciclo com apenas um local de terminação.
170  int i = 0;
171  while(i != length && first[i] == second[i])
172  i++;
173  return i == length;
174 }

176 // Esta função é usada durante a ordenação de _arrays_ de `double` realizada
177 // através do procedimento `qsort()` da biblioteca padrão do C. A utilização de
178 // ponteiros para `void` como tipo dos seus parâmetros é um requisito do
179 // procedimento `qsort()`, que, desta forma, é capaz de ordenar _arrays_ de
180 // itens de qualquer tipo. O procedimento `qsort()` requer que esta função
181 // devolva um valor inteiro que indica a relação de ordem entre o primeiro e o
182 // segundo argumento. Se o primeiro for superior ao segundo, o valor `int`
183 // devolvido deverá ser positivo. Se for igual, o valor devolvido deverá ser
184 // nulo. Se for menor, deverá devolver um valor negativo. Sendo esta função de
185 // utilidade restrita, optámos por lhe dar ligação local, usando para isso o
186 // qualificador `static`.
187 static int compare(const void *first_generic, const void *second_generic)
188 {
189  // Os argumentos são ponteiros para os itens a comparar. Sendo o
190  // procedimento `qsort()` genérico, os parâmetros correspondentes são do
191  // tipo `void *`. Para aceder aos valores do tipo `double`, fazem-se
192  // coerções para o tipo ponteiro para `double`.
193  const double *first = first_generic;
194  const double *second = second_generic;
195 
196  if (first < second)
197  return -1;
198  else if(first > second)
199  return 1;
200  else
201  return 0;
202 }

223 double double_array_stddev(int length, const double items[length])
224 {
225  assert(length >= 0);
226  assert(length == 0 || items != NULL);
227 
228  // Esta implementação é simples, mas ineficaz. Ver mais abaixo uma
229  // implementação do cálculo do desvio padrão que não exige percorrer o
230  // _array_ duas vezes.
231  const double average = double_array_average(length, items);
232 
233  double sum = 0.0;
234 
235  for (int i = 0; i != length; i++)
236  sum += square_of(items[i] - average);
237 
238  return sqrt(sum / length);
239 }

241 // Usamos uma forma de cálculo muito ineficaz: ordenar o _array_ de modo
242 // a obter a mediana através do ou dos valores centrais. Há formas bem
243 // mais eficazes de calcular a mediana. Nomeadamente, pode-se usar uma
244 // versão modificada do algoritmo de ordenação rápida que não se
245 // preocupa em ordenar o _array_, mas apenas em realizar
246 // particionamentos sucessivos até determinar o valor do ou dos itens
247 // centrais do _array_, que não precisa por isso de ser totalmente
248 // ordenado.
249 double double_array_median(int length, const double items[length])
250 {
251  assert(length >= 0);
252  assert(length == 0 || items != NULL);
253 
254  // Primeiro lidamos com o caso especial da mediana de zero itens.
255  if (length == 0)
256  return NAN;
257 
258  // Criamos um _array_ de trabalho que é uma cópia do _array_ original,
259  // pois este não deve ser alterado.
260  double work_items[length];
261  copy_double_array(length, work_items, items);
262 
263  // Usamos o procedimento de ordenação genérico do C `qsort()`, que
264  // utiliza o algoritmo de ordenação rápida, para ordenar o _array_ de
265  // trabalho.
266  qsort(work_items, length, sizeof(double), compare);
267 
268  // Se o _array_ tiver um número par de itens, a sua mediana é a média
269  // aritmética dos valores dos dois itens mais próximos do meio do
270  // _array_ ordenado. Se tiver um número ímpar de itens, então a mediana
271  // é o valor do item central do _array_ ordenado.
272  if (length % 2 == 0)
273  return (work_items[length / 2 - 1] + work_items[length / 2]) / 2;
274  else
275  return work_items[length / 2];
276 }

278 // Note que esta função está definida mesmo para zero itens, devolvendo infinito
279 // (+∞) nesse caso, pois o infinito é o elemento neutro da operação de obtenção do
280 // menor de dois operandos, da mesma forma que o zero (0) é o elemento neutro da
281 // operação de obtenção da soma de dois operandos.
282 double double_array_minimum(int length, const double items[length])
283 {
284  assert(length >= 0);
285  assert(length == 0 || items != NULL);
286 
287  double minimum = INFINITY;
288 
289  for (int i = 0; i != length; i++)
290  if (items[i] < minimum)
291  minimum = items[i];
292 
293  return minimum;
294 }

296 // Note que esta função está definida mesmo para zero itens, devolvendo infinito
297 // negativo (-∞) nesse caso, pois o infinito negativo é o elemento neutro da
298 // operação de obtenção do maior de dois operandos, da mesma forma que o zero
299 // (0) é o elemento neutro da operação de obtenção da soma de dois operandos.

303  assert(length == 0 || items != NULL);
304

306 
307  for (int i = 0; i != length; i++)

310 
311  return maximum;
312 }

314 struct double_statistics double_array_statistics(const int length,
315  const double items[length])
316 {
317  assert(length >= 0);
318  assert(length == 0 || items != NULL);
319 
320  // Esta variável guarda os valores das estatísticas e será devolvida
321  // quando a função retornar. Note a forma de inicialização dos campos da
322  // estrutura.
323  struct double_statistics statistics = {
324  .average = NAN,
325  .stddev = NAN,
326  .median = NAN,
327  .minimum = INFINITY,
328  .maximum = -INFINITY
329  };
330 
331  // Lidamos primeiro com o caso especial em que há zero itens apenas. O
332  // valor inicial da variável `statistics` foi escolhido de modo a cobrir
333  // este caso.
334  if (length == 0)
335  return statistics;
336 
337  // Uma vez que se calculam as várias estatísticas numa única função,
338  // podemos fazer algumas optimizações. Ainda assim, seria possível
339  // melhorar francamente a forma de cálculo da mediana, como referido
340  // anteriormente.
341  double work_items[length];
342 
343  double sum = 0.0;
344  double sum_of_squares = 0.0;
345 
346  for (int i = 0; i != length; i++) {
347  work_items[i] = items[i];
348  sum += items[i];
349  sum_of_squares += square_of(items[i]);
350  }
351 
352  statistics.average = sum / length;
353  statistics.stddev =
354  sqrt(sum_of_squares / length - square_of(statistics.average));
355 
356  qsort(work_items, length, sizeof(double), compare);
357 
358  statistics.minimum = work_items[0];
359 
360  if (length % 2 == 0)
361  statistics.median =
362  (work_items[length / 2 - 1] + work_items[length / 2]) / 2;
363  else
364  statistics.median = work_items[length / 2];
365 
366  statistics.maximum = work_items[length - 1];
367 
368  return statistics;
369 }
