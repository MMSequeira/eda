80 // Expressa em _clocks_, ou seja, em tiques do relógio. Esta constante indica o
81 // número mínimo de tiques de relógio a usar nas medições. Na máquina em que
82 // este programa foi testado, o relógio (tal como medido pela função `clock()`)
83 // tem um valor de `CLOCKS_PER_SEC` de 1&thinsp;000&thinsp;000. Logo, nenhuma
84 // medição será feita com intervalos de tempo inferiores a 1 segundo. Em
85 // ambientes onde o valor de `CLOCKS_PER_SEC` seja inferior, poderá ser
86 // necessário baixar a precisão, sob pena de as experiências se tornarem
87 // demasiado demoradas.
88 //
89 // Quando o tempo de execução de um algoritmo for inferior à precisão dada por
90 // esta variável, serão realizadas tantas execuções quantas necessárias para
91 // exceder este tempo, sendo o tempo de cada uma das execuções estimado através
92 // do quociente entre o tempo total das execuções sucessivas e o número de
93 // execuções realizadas (subtraindo-se depois o tempo necessário para, através
94 // de uma cópia, colocar no seu estado inicial, antes de cada ordenação, o
95 // _array_ a ordenar). Estas execuções sucessivas destinam-se a lidar com
96 // questões de resolução do relógio usado. Não as confunda com as repetições
97 // efectuadas com fins estatísticos, pois estas lidam com as flutuações do tempo
98 // de execução devido às condições de carga da máquina e outros efeitos com
99 // origem externa ao programa.
100 static const clock_t clock_precision = 1000000U; // clocks (tics)
101 
102 // Quando o tempo acumulado das repetições das estimativas do tempo de execução
103 // da ordenação, repetições essas realizadas para fins estatísticos, exceder
104 // este limiar, as repetições são interrompidas. Este limiar está expresso em
105 // segundos e tenta garantir que as experiências não se tornam demasiado
106 // demoradas. Naturalmente, as estatísticas obtidas serão tão piores quanto
107 // menor for o número de repetições realizadas. Um valor de 300 limita as
108 // repetições a cerca de cinco minutos.
109 static const double threshold_repetition_time = 300.0; // seconds
110 
111 // O número de repetições a efectuar para fins estatísticos é também limitado
112 // pelo valor desta constante.
113 static const int maximum_number_of_repetions = 1001;
114 
115 // Se o tempo de execução de um dado algoritmo se tornar superior a este limiar
116 // para uma dada dimensão do _array_ a ordenar, esse algoritmo será excluído das
117 // experiências realizadas com _arrays_ de maiores dimensões. Tenta-se assim
118 // impedir que algoritmos particularmente maus impeçam a realização de
119 // experiências com algoritmos melhores, que são capazes de lidar em tempo
120 // «útil» com _arrays_ de maiores dimensões.
121 static const double threshold_time_per_sort = 300.0; // seconds

200 // Função que devolve uma estimativa do tempo necessário para copiar o conteúdo
201 // do _array_ `items` para o _array_ `work_items`. Ambos os _arrays_ têm de ter
202 // pelo menos `length` itens. O tempo é devolvido em segundos. A estimativa é
203 // realizada com uma precisão dada pela constante `clock_precision`.
204 static double copy_time_estimate(const int length, double work_items[length],
205  const double items[length])
206 {
207  assert(length > 0);
208  assert(work_items != NULL);
209  assert(items != NULL);
210 
211  printf("\tEstimating copy time...\n");
212 
213  // Um primeiro ciclo estima o número de execuções da cópia necessárias
214  // para atingir a precisão pretendida.
215  int runs = 0;
216  clock_t start = clock();
217 
218  do {
219  copy_double_array(length, work_items, items);
220  runs++;
221  } while(clock() - start < clock_precision);
222 
223  // Uma vez que a obtenção da estimativa do número de execuções teve de
224  // recorrer a invocações repetidas da função `clock()`, o que poderia
225  // viciar uma estimativa do tempo de cópia que se baseasse no tempo de
226  // acumulado da execução do ciclo, repetimos o mesmo número de
227  // execuções, desta vez sem invocações intermédias à função `clock()`.
228  start = clock();
229 
230  for (int i = 0; i != runs; i++)
231  copy_double_array(length, work_items, items);
232 
233  double copy_time = (double) (clock() - start) / CLOCKS_PER_SEC / runs;
234 
235  printf("\tCopy time estimated to be %g seconds (based on %d runs).\n",
236  copy_time, runs);
237 
238  return copy_time;
239 }

241 // Função que devolve uma estimativa do tempo de execução do algoritmo de
242 // ordenação dado por `algorithm`, quando aplicado à ordenação do _array_
243 // `work_itens` com `length` itens, obtido por cópia do _array_ `items` com o
244 // mesmo número de itens. A estimativa é realizada recorrendo à acumulação do
245 // tempo de `runs` execuções sucessivas. Uma vez que cada execução do algoritmo
246 // tem de ser precedida da cópia dos itens do _array_ `items` para o _array_
247 // `work_items`, o tempo estimado para essa cópia, `copy_time`, é subtraído ao
248 // quociente entre o tempo total acumulado e o número de execuções `runs`. Em
249 // caso de erro, devolve um valor negativo.
250 static double sort_time_estimate(struct sorting_algorithm algorithm,
251  const int length, double work_items[length],
252  const double items[length], const int runs,
253  const double copy_time)
254 {
255  assert(length > 0);
256  assert(work_items != NULL);
257  assert(items != NULL);
258  assert(runs > 0);
259  assert(copy_time > 0.0);
260 
261  clock_t start = clock();
262 
263  for (int i = 0; i != runs; i++) {
264  copy_double_array(length, work_items, items);
265 
266  if(algorithm.sort(length, work_items)) {
267  fprintf(stderr, "Error: could not run sorting "
268  "algorithm '%s'.\n", algorithm.name);
269  return -1.0;
270  }
271  }
272 
273  return (double) (clock() - start) / CLOCKS_PER_SEC / runs - copy_time;
274 }

276 // Devolve uma estimativa do número de execuções do algoritmo de ordenação
277 // `algorithm`, quando aplicado à ordenação do _array_ `work_itens` com `length`
278 // itens, obtido por cópia do _array_ `items` com o mesmo número de itens,
279 // necessárias para atingir a precisão pretendida. O efeito do tempo de cópia
280 // entre os _arrays_ `copy_time` é cancelado na estimativa efectuadas. Em caso
281 // de erro devolve um valor negativo.
282 static int number_of_runs(const struct sorting_algorithm algorithm,
283  const int length, double work_items[length],
284  const double items[length], const double copy_time)
285 {
286  assert(length > 0);
287  assert(work_items != NULL);
288  assert(items != NULL);
289  assert(copy_time > 0.0);
290 
291  int runs = 0;
292  clock_t start = clock();
293 
294  do {
295  copy_double_array(length, work_items, items);
296 
297  if(algorithm.sort(length, work_items)) {
298  fprintf(stderr, "Error: could not run sorting "
299  "algorithm '%s'.\n", algorithm.name);
300  return -1;
301  }
302 
303  runs++;
304  } while(clock() - start - runs * copy_time * CLOCKS_PER_SEC <
305  clock_precision);
306 
307  return runs;
308 }

310 // Esta rotina escuta uma experiência com o algoritmo de ordenação `algorithm`,
311 // obtendo contagens de operações e estatísticas do tempo de execução
312 // resultantes da sua aplicação ao _array_ `items`, com comprimento `length`. As
313 // estatísticas são optidas por aplicação repetida da função
314 // `sort_time_estimate()`. O _array_ ordenado é sempre `work_itens`, para onde
315 // os itens originais, contidos em `items`, são copiados antes de cada
316 // ordenação. Para além da obtenção de contagens e estatísticas, a correcção da
317 // ordenação obtida é verificada por comparação com o _array_ `sorted_items`,
318 // que contém os mesmos itens de `items`, mas já ordenados por ordem crescente.
319 // As estimativa dos tempo de execução são realizadas levando em conta o tempo
320 // estimado `copy_time` necessário para preparar o _array_ `sorted_items` para
321 // cada ordenação, copiando os seus itens a partir de `items`. As contagens e
322 // estatísticas são guardadas na instância de `struct algorithm_statistics`
323 // apontada pelo ponteiro `statistics`. Em caso de erro devolve o valor `true`.
324 static bool experiment_algorithm(const struct sorting_algorithm algorithm,
325  const int length, double work_items[length],
326  const double items[length],
327  const double sorted_items[length],
328  const double copy_time,
329  struct algorithm_statistics *statistics)
330 {
331  assert(length > 0);
332  assert(work_items != NULL);
333  assert(items != NULL);
334  assert(sorted_items != NULL);
335  assert(copy_time > 0.0);
336  assert(statistics != NULL);
337 
338  // Preparamos o _array_ `work_items` e realizamos uma primeira ordenação
339  // invocando o procedimento de ordenação que implementa o algoritmo em
340  // causa e que efectua contagem das operações elementares. Dessa forma,
341  // não só obtemos as contagens necessárias, como podemos verificar a
342  // correcção do resultado da ordenação com essa função, comparando-o com
343  // o _array_ `sorted_items`. Em caso de erro, retornamos devolvendo o
344  // valor `true`.
345  copy_double_array(length, work_items, items);
346 
347  printf("\t\tRunning counting algorithm version.\n");
348 
349  if (algorithm.sort_and_count(length, work_items, &statistics->counts)) {
350  fprintf(stderr, "Error: could not run sorting algorithm '%s' "
351  "(counting version).\n", algorithm.name);
352  return true;
353  }
354 
355  printf("\t\tChecking correctness of counting algorithm version.\n");
356 
357  if (!double_arrays_equal(length, sorted_items, work_items)) {
358  fprintf(stderr, "Error: sorting algorithm '%s' (counting "
359  "version) did not sort.\n", algorithm.name);
360  return true;
361  }
362 
363  // Antes de proceder às estimativas dos tempos de execução e ao cálculo
364  // das respectivas estatísticas, voltamos a preparar o _array_
365  // `work_items` e aplicamos-lhe a versão do procedimento de ordenação
366  // que implementa o algoritmo de ordenação em causa _sem efectuar
367  // contagens_. Fazemo-lo para podermos verificar a correcção do
368  // resultado obtido, da mesma forma que anteriormente. Em caso de erro,
369  // retornamos devolvendo o valor `true`.
370  copy_double_array(length, work_items, items);
371 
372  printf("\t\tRunning non-counting algorithm version.\n");
373 
374  if(algorithm.sort(length, work_items)) {
375  fprintf(stderr, "Error: could not run sorting algorithm '%s'.\n",
376  algorithm.name);
377  return true;
378  }
379 
380  printf("\t\tChecking correctness of non-counting algorithm version.\n");
381 
382  if (!double_arrays_equal(length, sorted_items, work_items)) {
383  fprintf(stderr, "Error: sorting algorithm '%s' did not sort.\n",
384  algorithm.name);
385  return true;
386  }
387 
388  // Feitas as contagens e verificada a correcção das ordenações obtidas
389  // por aplicação dos dois procedimentos implementando o algoritmo,
390  // podemos começar a estimar os tempos de execução. Estes tempos são
391  // estimados usando a versão _sem contagem_ dos procedimentos.
392  printf("\t\tTime measurements:\n");
393 
394  // Estimamos o número de execuções da ordenação a realizar para obter
395  // resultados com a precisão escolhida. Em caso de erro, retornamos
396  // devolvendo o valor `true`.
397  int runs =
398  number_of_runs(algorithm, length, work_items, items, copy_time);
399 
400  if (runs < 0)
401  return true;
402 
403  printf("\t\t\tEach time will be estimated using %d runs.\n", runs);
404 
405  // Guardamos o valor obtido na estrutura de estatísticas.
406  statistics->accumulated_runs = runs;
407 
408  // Os tempos obtidos em cada uma das repetições das estimativas a
409  // realizar serão guardados neste _array_.
410  double times[maximum_number_of_repetions];
411 
412  printf("\t\t\tStarting repeated estimations:\n");
413 
414  // Este ciclo repete a obtenção de estimativas do tempo de execução até
415  // se atingir o limite máximo de iterações ou até se ultrapassar o
416  // limiar do tempo acumulado de execução do ciclo.
417  int repetitions = 0;
418  clock_t start = clock();
419  double accumulated_time;
420 
421  do {
422  // Obtemos uma estimativa do tempo de execução do algoritmo. Em
423  // caso de erro, retornamos devolvendo o valor `true`.
424  times[repetitions] =
425  sort_time_estimate(algorithm, length, work_items,
426  items, runs, copy_time);
427 
428  if(times[repetitions] < 0.0)
429  return true;
430 
431  repetitions++;
432 
433  accumulated_time = (double)(clock() - start) / CLOCKS_PER_SEC;
434  } while(repetitions != maximum_number_of_repetions &&
435  accumulated_time < threshold_repetition_time);
436 
437  printf("\t\t\t%d repetitions in %g seconds.\n", repetitions,
438  accumulated_time);
439 
440  // Guardamos o valor obtido para as repetições na estrutura de
441  // estatísticas.
442  statistics->repetitions = repetitions;
443 
444  // Calculamos as estatísticas dos tempos de execução obtidos e guardamo-
445  // las na estrutura de estatísticas.
446  statistics->times = double_array_statistics(repetitions, times);
447 
448  // Retornamos devolvendo o valor `false`, indicando que não ocorreram
449  // quaisquer erros.
450  return false;
451 }

498 // Escrevemos os resultados no canal de saída `output`. Se algum dos algoritmos
499 // tiver um tempo de execução superior ao limiar definido, actualizamos o
500 // _array_ `excessive_time_per_sort` de modo a conter o valor `true` na posição
501 // correspondente a esse algoritmo. Se alguma posição desse _array_ já tiver o
502 // valor `true`, o algoritmo correspondente não chega a ser experimentado.
503 // Devolvemos `true` em caso de erro.
504 bool experiment_size(FILE *const output, const char *const path,
505  const char *const file_type, const int size,
506  bool excessive_time_per_sort[number_of_sorting_algorithms])
507 {
508  // Construímos o nome do ficheiro com o tipo dado por `file_type`, com a
509  // dimensão dada por `s` e na pasta dada por `path`. Este ficheiro
510  // contém os itens a ordenar.
511  char file_name[FILENAME_MAX];
512  snprintf(file_name, FILENAME_MAX, "%s%s_%d.txt", path, file_type, size);
513 
514  // Construímos o nome do ficheiro com o tipo `sorted`, com a dimensão
515  // `s` e na pasta dada por `path`. Este ficheiro contém os itens já
516  // ordenados. O seu conteúdo será usado para verificar a correcção dos
517  // algoritmos.
518  char sorted_file_name[FILENAME_MAX];
519  snprintf(sorted_file_name, FILENAME_MAX, "%ssorted_%d.txt", path, size);
520 
521  // Definimos os ponteiros que apontarão para os _arrays_ dinâmicos
522  // necessários para a experiência. Violamos aqui explicitamente a regra
523  // de definir as variáveis tão próximo quanto possível do primeiro local
524  // em que podem ser inicializadas com um valor relevante. Essa violação
525  // decorre do estilo de tratamento de erros usado, que recorre à
526  // geralmente pouco recomendável instrução `goto` para se poder «arrumar
527  // a casa», em caso de erro, de forma pouco intrusiva no restante
528  // código.
529  double *items = NULL;
530  double *sorted_items = NULL;
531  double *work_items = NULL;
532 
533  // Lemos o conteúdo do ficheiro com os itens a ordenar para o _array_
534  // dinâmico `items` (na realidade um ponteiro para o seu primeiro item).
535  // O número de itens lidos fica guardado em `length`. A leitura e
536  // criação do _array_ dinâmico podem falhar por falta de memória ou por
537  // o ficheiro não conter o número de itens esperado.
538  int length;
539  items = read_double_array_from(file_name, &length);
540 
541  // Definimos a variável `error`, cujo valor indica, em cada momento, se
542  // ocorreu algum erro. O estilo de tratamento de erros que adoptámos
543  // inspira-se no utilizado no _kernel_ do Linux. Ou seja, os erros são
544  // detectados logo que possível e levam a um salto, usando a instrução
545  // `goto`, para a secção final e de «arrumação da casa» da rotina em que
546  // nos encontramos. Dessa forma o código fica mais simples e
547  // compreensível. Em linguagens de mais alto nível que o C há melhores
548  // mecanismos para resolver este problema de lidar com os erros de forma
549  // elegante.
550  //
551  // Inicializamos a variável `error` com o valor apropriado, dado o
552  // resultado da tentativa de leitura do ficheiro efectuada na instrução
553  // anterior.
554  bool error = items == NULL || length != size;
555 
556  // Em caso de erro escrevemos em `stderr` uma mensagem apropriada e
557  // saltamos para a secção de tratamento de erros, a que daremos o nome
558  // `terminate`.
559  if (error) {
560  fprintf(stderr, "Error: Reading file '%s'.\n", file_name);
561  goto terminate;
562  }
563 
564  // Lemos o conteúdo do ficheiro com os itens já ordenados para o _array_
565  // dinâmico `sorted_items` (na realidade um ponteiro para o seu primeiro
566  // item). O número de itens lidos fica guardado em `length`. A leitura e
567  // criação do _array_ dinâmico podem falhar por falta de memória ou por
568  // o ficheiro não conter o número de itens esperado. Em caso de falha, o
569  // erro é tratado como anteriormente.
570  sorted_items = read_double_array_from(sorted_file_name, &length);
571 
572  error = sorted_items == NULL || length != size;
573 
574  if (error) {
575  fprintf(stderr, "Error: Reading file '%s'.\n", sorted_file_name);
576  goto terminate;
577  }
578 
579  // Construímos o _array_ de trabalho, ou seja, o _array_ dinâmico para
580  // onde serão copiados os itens a ordenar sempre que necessário e que
581  // será ordenado durante as experiências a realizar. O _array_ `items`
582  // não será alterado. A criação deste _array_ pode falhar por falta de
583  // memória. Criar este _array_ neste local evita a necessidade de o
584  // recriar várias vezes durante as experiências. A criação do _array_
585  // pode falhar por falta de memória.
586  work_items = new_double_array_of(size);
587 
588  error = work_items == NULL;
589 
590  if (error) {
591  fprintf(stderr, "Error: Allocating work array.\n");
592  goto terminate;
593  }
594 
595  // Estimamos o tempo de execução da cópia dos itens do _array_ `items`
596  // para o _array_ `work_items`. Este tempo será descontado das
597  // estimativas do tempo de execução dos algoritmos, que sem este
598  // desconto incluiriam o tempo demorado a copiar os itens para o _array_
599  // de trabalho antes de proceder à sua ordenação.
600  double copy_time = copy_time_estimate(length, work_items, items);
601 
602  // Escrevemos no ficheiro CSV de resultado o valor da primeira coluna,
603  // ou seja, a dimensão do ficheiro em ordenação. Ou, o que é o mesmo, o
604  // número de itens no _array_ a ordenar.
605  fprintf(output, "%d", size);
606 
607  // Percorremos cada um dos algoritmos a experimentar.
608  for (int a = 0; a != number_of_sorting_algorithms; a++) {
609  // Inicializamos as estatísticas com os valores iniciais.
610  struct algorithm_statistics statistics = initial_statistics;
611 
612  printf("\tStarting experiments for %s:\n",
613  sorting_algorithms[a].name);
614 
615  // Se o tempo de execução do algoritmo corrente não tiver
616  // excedido anteriormente (i.e., para uma dimensão inferior do
617  // ficheiro a ordenar) o limiar estabelecido, então a
618  // experiência será realizada. Caso contrário, os dados
619  // guardados no ficheiro de resultado serão os valores guardados
620  // na constante `initial_statistics`.
621  if (!excessive_time_per_sort[a]) {
622  printf("\t\tPerforming experiments.\n");
623  // Realizamos a experiência invocando a rotina de
624  // experimentação e verificando se ocorreu algum erro
625  // durante o processo. A experiência é realizada com o
626  // algoritmo `sorting_algorithm[a]`, usando os _arrays_
627  // `work_items`, `items` e `sorted_items`, todos com
628  // `length` itens. O tempo de cópia `copy_time` é usado
629  // para efectuar descontos nos tempos estimados sempre
630  // que necessário. Os resultados são guardados na
631  // variável `statistics`, cujo endereço é passado à
632  // rotina de experimentação. A rotina de experimentação
633  // devolve um valor booleano que indica se ocorreu ou
634  // não algum erro.
635  error = experiment_algorithm(sorting_algorithms[a],
636  size, work_items, items,
637  sorted_items, copy_time,
638  &statistics);
639  if (error)
640  goto terminate;
641  } else
642  printf("\t\tSkipping experiments.\n");
643 
644  printf("\t\tWriting results, if any.\n");
645 
646  // Escrevemos os resultados da experiência, caso esta se tenha
647  // realizado. Caso contrário, escreve-se o valor inicial das
648  // estatísticas, que assinalarão a experiência como não
649  // realizada.
650  write_statistics(output, statistics);
651 
652  printf("\t\tEnding experiments for %s (median time = %g s).\n",
653  sorting_algorithms[a].name, statistics.times.median);
654 
655  // Actualizamos o _array_ que indica se o tempo de ordenação
656  // excedeu em algum momento, para o algoritmo corrente, o limiar
657  // estabelecido.
658  if (statistics.times.median > threshold_time_per_sort)
659  excessive_time_per_sort[a] = true;
660  }
661 
662  // Terminamos a linha de estatísticas escrita no ficheiro CSV de
663  // resultado.
664  fputc('\n', output);
665 
666  // A secção final é dedicada a «arrumar a casa», libertando a memória
667  // reservada. Esta secção pode ser executada tanto no caso de não terem
668  // ocorrido erros, como no caso de terem ocorrido erros. Neste último
669  // caso, pode acontecer que o erro tenha ocorrido antes da criação dos
670  // _arrays_ dinâmicos. O código está construído de tal forma que, se
671  // isso acontecer, o correspondente ponteiro terá o valor `NULL`, que
672  // pode ser passado sem inconveniente à rotina `free()`.
673 terminate:
674  // Libertamos a memória reservada para cada um dos _arrays_
675  // dinâmicos.
676  free(work_items);
677  free(sorted_items);
678  free(items);
679 
680  // Retornamos devolvendo o valor na variável `error`, ou seja, indicando
681  // se ocorreram ou não erros.
682  return error;
683 }
684 
685 // A rotina principal do programa, que executa as experiências para cada
686 // dimensão dos ficheiros com valores a ordenar e para cada algoritmo.
687 // Efectuamos as experiências apenas para os ficheiros do tipo dado por
688 // `file_type` (que pode tomar os valores `sorted`, `partially_sorted` e
689 // `shuffled`). Lemos os ficheiros a partir da pasta dada por `path` (que tem de
690 // terminar no caractere separador de pastas correspondente ao sistema operativo
691 // em que o programa é executado). Escrevemos os resultados no ficheiro com nome
692 // dado por `statistics_file_name`. Em caso de erro devolvemos o valor `true`.
693 static bool experiment_all(const char *const path,
694  const char *const file_type,
695  const char *const statistics_file_name)
696 {
697  assert(path != NULL);
698  assert(file_type != NULL);
699  assert(statistics_file_name != NULL);
700 
701  // Definimos um _array_ de valores booleanos que indicam se, em alguma
702  // das experiências já realizada com o correspondente algoritmo se
703  // excedeu o limiar do tempo de ordenação. Os algoritmos para os quais
704  // isso aconteça deixam de ser usados com ficheiros de maiores
705  // dimensões, para evitar que o tempo de execução do programa se torne
706  // demasiado grande.
707  bool excessive_time_per_sort[number_of_sorting_algorithms];
708  for (int a = 0; a != number_of_sorting_algorithms; a++)
709  excessive_time_per_sort[a] = false;
710 
711  // Abrimos o canal para o ficheiro de resultados. O ficheiro de
712  // resultados tem o formato CSV (comma separated values) e usa o
713  // caractere «;» como separador. A sua primeira linha contém a
714  // identificação dos valores em cada coluna, ou seja, os cabeçalhos
715  // correspondentes a cada uma das colunas. As linhas subsequentes têm os
716  // valores obtidos experimentalmente, correspondendo cada linha a uma
717  // dada dimensão dos ficheiros de entrada. A primeira coluna contém essa
718  // dimensão. As colunas subsequentes contém as estatísticas obtidas para
719  // cada um dos algoritmos experimentados.
720  FILE *const output = fopen(statistics_file_name, "w");
721 
722  // Definimos a variável `error`, que indicará em cada instante se até
723  // então ocorreu algum erro, e inicializamo-la verificando se a abertura
724  // do canal de escrita teve resultado.
725  bool error = output == NULL;
726 
727  if (error) {
728  fprintf(stderr, "Error: Could not open '%s' for writing!\n",
729  statistics_file_name);
730  goto terminate;
731  }
732 
733  // Escrevemos o cabeçalho da primeira coluna, que contém a dimensão dos
734  // ficheiros usados para obter as estatísticas de cada linha.
735  fprintf(output, "Size");
736 
737  // Escrevemos os cabeçalhos correspondentes a cada um dos algoritmos.
738  for (int a = 0; a != number_of_sorting_algorithms; a++)
739  write_statistics_headers(output, sorting_algorithms[a].name);
740 
741  // Terminamos a linha de cabeçalho.
742  fputc('\n', output);
743 
744  printf("Wrote headers to results.\n");
745 
746  // Percorremos todas as dimensões de ficheiros, que são potências de 2
747  // entre 2 e 2<sup>24</sup>. Este último valor é dado pela constante
748  // `maximum_file_size`. A operação realizada na guarda serve para obter
749  // o valor 2<sup>25</sup>, valor para o qual a se tornará falsa. O
750  // operador `<<` desloca o padrão de _bits_ guardado numa variável
751  // inteira de um _bit_ para esquerda, o que corresponde a uma
752  // multiplicação por 2.
753  for (int size = 1 << 1; size != maximum_file_size << 1; size <<= 1) {
754  printf("Starting experiments for size %d:\n", size);
755 
756  // Invocamos a rotina que executa as experiências para a
757  // dimensão `size` dos ficheiros e para todos os algoritmos.
758  error = experiment_size(output, path, file_type, size,
759  excessive_time_per_sort);
760  if (error)
761  goto terminate;
762 
763  printf("\tEnded experiments for size %d.\n", size);
764  }
765 
766  // Secção de «arrumação da casa».
767 terminate:
768  // Fechamos o canal para o ficheiro CSV de resultado.
769  if (output != NULL)
770  fclose(output);
771 
772  // Retornamos devolvendo o valor na variável `error`, ou seja, indicando
773  // se ocorreram ou não erros.
774  return error;
775 }

777 // Rotina inicial do programa.
778 int main(const int argument_count,
779  const char *const argument_values[argument_count])
780 {
781  // Os argumentos recebidos através da linha de comandos, incluindo o
782  // próprio nome do programa executável, são quatro: esse nome, a pasta
783  // onde os ficheiros a ordenar se encontram, o tipo de ficheiros a
784  // ordenar e o nome do ficheiro onde os resultados serão escritos no
785  // formato CSV, por esta ordem.
786  if (argument_count < 4) {
787  fprintf(stderr, "Error: Insuficient number of arguments!\n");
788  return EXIT_FAILURE;
789  }
790 
791  const char *const path = argument_values[1];
792  const char *const file_type = argument_values[2];
793  const char *const statistics_file_name = argument_values[3];
794 
795  // Verificamos a correcção do tipo de ficheiro passado na linha de
796  // comandos.
797  if (!valid_file_type(file_type)) {
798  fprintf(stderr, "Error: Unknown file type '%s'!\n", file_type);
799  return EXIT_FAILURE;
800  }
801 
802  printf("Starting experiments from %s files in %s. Storing results "
803  "in %s.\n", file_type, path, statistics_file_name);
804 
805  // Executamos o procedimento principal do programa, verificando se essa
806  // execução teve sucesso.
807  if (experiment_all(path, file_type, statistics_file_name))
808  return EXIT_FAILURE;
809 
810  // Terminamos assinalando sucesso.
811  return EXIT_SUCCESS;
812 }
