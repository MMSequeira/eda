13 // Os _arrays_ são uma das formas de agregar dados em C. Nos _arrays_ os dados,

19 // ```

21 //
22 // ### Inicialização de _arrays_
23 //
24 // #### Inicialização por omissão
25 //
26 // Tal como acontece no caso de variáveis dos tipos básicos da linguagem, por
27 // omissão o C _não_ inicializa os itens de um _array_ (desde que seja um
28 // _array_ automático, pois se for estático todos os itens são inicializados com
29 // um valor nulo). Ou seja, estes começam a sua vida com valores que, do ponto
30 // de vista do programador, são puro lixo, mesmo que provavelmente estes valores
31 // consistam simplesmente numa interpretação do padrão de _bits_ que se encontra
32 // na memória alocada para esses itens ou variáveis.
33 //
34 // #### Inicialização explícita
35 //
36 // Os _arrays_ podem ser inicializados explicitamente através de uma lista
37 // de inicializadores:
38 //
39 // ```C
40 // int array[4] = {0, 1, 2};
41 // ```
42 //
43 // Neste caso omitimos o valor inicial do quarto item do _array_. Quando isto
44 // acontece, o compilador do C encarrega-se de gerar código máquina que
45 // inicializa os itens em falta com valores nulos. Podemos, por isso, usar um
46 // truque simpático para inicializar todos os itens dos _arrays_ do C com o
47 // valor zero (no entanto, a norma que especifica o C11 não é clara quanto à
48 // validade de usar uma lista de inicialização vazia...):
49 //
50 // ```C
51 // int array[4] = {};
52 // ```
53 //
54 // Quando se fornece uma lista de inicialização, é possível omitir a
55 // dimensão dos _arrays_. Nesse caso, o _array_ terá como dimensão o número
56 // de valores colocado entre chavetas. Por exemplo, o código
57 //
58 // ```C
59 // int array[] = {0, 1, 2};
60 // ```
61 //
62 // define um _array_ com exactamente três itens.

110 // Definição de uma função
111 // -----------------------
112 //
113 // Definimos uma função que devolve a média dos itens da matriz passada como
114 // argumento. Uma definição de uma rotina consiste num _cabeçalho_, onde se
115 // caracteriza a interface da rotina, e num _corpo_, onde se coloca a
116 // _implementação_ da rotina. Neste caso temos uma _definição_, e não uma
117 // simples _declaração_, pois incluímos o corpo da função. Todas as definições
118 // são também declarações, mas nem todas as declarações são definições. Uma
119 // declaração simples, sem indicação da implementação, inclui apenas o cabeçalho
120 // da rotina seguido de um ponto e vírgula. O ficheiro
121 // [`array_utils.h`](array_utils.html) contém apenas a declaração da função.
122 //
123 // ### Documentação e cabeçalho da função
124 //
125 // O _cabeçalho_ de uma rotina declara a sua existência e especifica a sua forma
126 // de utilização, mas não especifica totalmente a _interface_ da função. A
127 // interface completa de uma rotina inclui informação mais pormenorizada acerca
128 // dos objectivos da rotina, acerca do significado dos parâmetros, etc., mas,
129 // sobretudo, estabelece um contrato claro entre o _programador produtor_ da
130 // rotina, i.e., o ou os programadores que a implementam e mantêm, e o
131 // _programador consumidor ou cliente_ da rotina, i.e., o ou os programadores
132 // que a usam no seu código. O C não suporta directamente a especificação da
133 // interface completa da rotina. Por isso, essa especificação é feita recorrendo
134 // a um _comentário de documentação_.
135 
136 // #### Documentação
137 //
138 // O cabeçalho é precedido de um _comentário de documentação_ que indica
139 // claramente qual o contrato desta função. Usamos aqui o formato de
140 // documentação [Doxygen](doxygen.org).
141 //
142 // O comentário de documentação inclui:
143 //
144 // - Descrição breve da função.
145 // - Descrição mais longa da função (no final).
146 // - Descrição de cada parâmetro.
147 // - Descrição do valor devolvido.
148 // - Indicação das pré-condições (do contrato).
149 // - Indicação das pós-condições (do contrato).
166 //
167 // #### Cabeçalho
168 //
169 // Este cabeçalho inclui a seguinte informação:
170 //
171 // - Nome da função &ndash; O nome da função deve ser tão explícito e claro
172 // quanto possível. Pense numa possível utilização da função para chegar a um
173 // bom nome. Por exemplo, suponha-se que existe um _array_ chamado
174 // `student_grades` com a lista das notas de um dado aluno e se pretende saber
175 // se a sua média ultrapassa os 15 valores. O código a escrever seria o que se
176 // segue, que é legível quase como se se tratasse de inglês:
177 //
178 // ```C
179 // if (average_of(student_grades) > 15.0)
180 // ...
181 // ```
182 //
183 // - Tipo do valor devolvido &ndash; No caso de se tratar de um procedimento, o
184 // mais natural será usar `void`, e não devolver nada (embora em C seja
185 // bastante usual devolver um valor que permita lidar com potenciais erros).
186 // No caso de se tratar de uma função, será o tipo do valor a devolver quando
187 // esta retornar. Neste caso pretende-se devolver a média de um conjunto de
188 // itens inteiros. Como a média de números inteiros só esporadicamente será
189 // também um número inteiro, optou-se por um tipo que abarca valores
190 // racionais.
191 //
192 // - Tipo e nome de cada dos parâmetros que serão inicializados com o valor do
193 // argumento respectivo sempre que a rotina for invocada. Neste caso
194 // particular declaram-se dois parâmetros. O primeiro é um inteiro que será
195 // inicializado com o número dos itens do _array_ sobre os quais se deve
196 // devolver a média. O segundo é o _array_ que contém os itens cuja média será
197 // devolvida.
198 double average_of(int number_of_items, int items[number_of_items])
199 
200 // #### E no entanto...
201 //
202 // E no entanto, o número de itens no segundo parâmetro é ignorado pelo C. Isso
203 // mesmo: ignorado. Sempre que o C encontra uma declaração de um parâmetro como
204 // um _array_, ignora a dimensão do _array_. Ou seja, o cabeçalho usado é
205 // equivalente ao seguinte:
206 //
207 // ```C
208 // double average_of(int number_of_items, int items[])
209 // ```
210 //
211 // Sendo assim, como pode o compilador saber qual a dimensão do _array_? Bem,
212 // não pode! É que as transformações não ficam por aqui. O compilador, na
213 // realidade, não se limita a ignorar a dimensão do _array_. Na realidade altera
214 // a declaração de qualquer parâmetro que esteja uma _array_ num ponteiro para o
215 // tipo dos itens desse _array_. Ou seja, na realidade a declaração de um
216 // parâmetro como _array_ é _pura cosmética_. O que não é necessariamente mau,
217 // pois essa «cosmética» pode deixar mais clara a intenção do programador do que
218 // uma versão mais fiel à realidade. Será que prefere mesmo a versão fiel? Veja
219 // o resultado:
220 //
221 // ```C
222 // double average_of(int number_of_items, int* items)
223 // ```
224 //
225 // Nesta última versão não é de todo claro, excepto pelo facto de o nome do
226 // parâmetro estar no plural, de que se trata de um _array_ de itens.

228 // ### Corpo da função
229 //
230 // Tal como o cabeçalho de uma rotina é parte da especificação da sua interface,
231 // especificando parcialmente o seu «como se usa», o corpo de uma rotina é a sua
232 // implementação, definindo o seu «como funciona». O corpo de uma rotina
233 // estrutura-se usualmente como se segue:
234 //
235 // 1. Verificação de pré-condições. Verificam-se duas pré-condições, nesta
236 // função.
237 //
238 // 2. Resolução de casos especiais, por ordem decrescente de especificidade. Não
239 // se aplica nesta função.
240 //
241 // 3. Resolução do caso geral.
242 //
243 // 4. Verificação de pós-condições. No nosso caso não é prático verificar a pós-
244 // condição, infelizmente.
245 //
246 // 4. Retorno da rotina, com possível devolução de um valor. Neste caso retorna-
247 // se devolvendo a média pretendida.
248 
249 {
250  // Há três formas muito diferentes de lidar com as violações das pré-
251  // condições.
252  //
253  // A primeira forma, usada aqui, passa por fazer asserções acerca das
254  // pré- condições. As asserções são proposições que se afirma serem
255  // verdadeiras acerca do estado do programa. Se as asserções forem
256  // realizadas recorrendo à _macro_ `assert()`, então o programa abortará
257  // no caso de alguma das proposições testada não ser verdadeira. Por
258  // exemplo, na função em definição afirma-se que o primeiro argumento
259  // não pode ser negativo e que o segundo não pode ser um ponteiro nulo.
260  // Embora estas afirmações não esgotem as pré- condições da função, são
261  // as que é possível ou prático fazer. Tal como no caso das pós-
262  // condições, é um bom exercício tentar cobrir as pré-condições, tanto
263  // quanto possível ou prático, com asserções. Que acontece quando uma
264  // das asserções desta função, a primeira, por exemplo, falha? Para além
265  // de o programa abortar, é mostrada uma mensagem de erro apropriada:
266  //
267  // <pre>
268  // Assertion failed: (number_of_items >= 0), function average_of,
269  // file /Users/mmsequeira/Projects/eda/arrays_basics/arrays_basics.c,
270  // line 311.
271  // </pre>
272  //
273  // Esta primeira forma de lidar com violações de pré-condições é também
274  // a única aceitável no caso da violação de pós-condições, uma vez o
275  // erro é detectado pelo mesmo troço de código que o causou, não havendo
276  // por isso qualquer possibilidade de recuperação: trata-se sempre do
277  // programador produtor da rotina em causa.
278  //
279  // A segunda forma de lidar com violações de pré-condições reconhece que
280  // a responsabilidade por estas violações não é do programador produtor
281  // da rotina, mas sim do programador cliente, sendo por isso boa ideia
282  // dar a esse programador uma possibilidade de recuperar dos seus
283  // próprios erros. Para o permitir, a rotina, ao detectar a violação,
284  // retorna imediatamente, devolvendo um valor especial que indica ter
285  // ocorrido um erro. No caso de um procedimento, que sem esta
286  // preocupação não devolveria nada, usando `void` como tipo de
287  // devolução, é fácil e convencional lidar com erros: o tipo de
288  // devolução passa a `int` e a devolução que um qualquer valor não nulo
289  // significa que ocorreu um erro. O valor exacto devolvido pode ser
290  // usado como um código permitindo identificar o tipo de erro ocorrido.
291  // A devolução de 0 tem o significado oposto: significa que o
292  // procedimento foi executado com sucesso. No caso de uma função, esta
293  // solução só pode ser usada se o conjunto imagem da função (o conjunto
294  // dos valores que a função pode devolver em caso de sucesso) não
295  // esgotar o seu contra-domínio (dado pelo tipo de devolução). Ou seja,
296  // pode ser usado se a função _não for sobrejectiva_ e, por isso, se
297  // podem usar valores que não pertençam à imagem da função como códigos
298  // de erro. Por exemplo, uma função de calcule e devolva um comprimento
299  // de uma sequência tem o tipo `int` com contradomínio mas como imagem
300  // tem apenas os inteiros não negativos, pelo que qualquer valor
301  // negativo pode ser usado como código de erro. Esta segunda forma de
302  // lidar com os erros não é possível no caso desta função, pois qualquer
303  // valor do tipo `double` é média de um número indefinido de possíveis
304  // _arrays_.
305  //
306  // A terceira forma de lidar com erros passa por usar uma variável
307  // global para registar um código do erro ocorrido. Embora usada, esta
308  // forma de assinalar erros é muito perigosa, tal como qualquer programa
309  // que recorra a variáveis globais o é. Para mais informação estude o
310  // ficheiro de cabeçalho da biblioteca padrão
311  // [`errno.h`](http://en.wikipedia.org/wiki/Errno.h).
312  assert(number_of_items >= 0);
313  assert(items != NULL);
314 
315  // A variável `sum` é um _acumulador_ que guarda, em cada passo do
316  // ciclo, a soma dos primeiros `i` itens do _array_ `items` (note como,
317  // mesmo sabendo que `items` é um ponteiro para o primeiro item de um
318  // _array_, e não um _array_ propriamente dito, continuamos a chamar-lhe
319  // _array_). Uma vez que a acumulação a realizar é uma soma, o
320  // acumulador é inicializado com o elemento neutro da soma: 0 (zero).
321  double sum = 0.0;
322 
323  // O ciclo percorre os primeiros `number_of_items` itens do _array_
324  // `items`, actualizando as variáveis `sum` e `i` de forma a que `sum`
325  // se mantenha sempre como a soma dos primeiros `i` items de `items` e
326  // de forma a que `i` progrida um a um, desde o valor inicial até
327  // atingir o valor `número_de_itens`.
328  for (int i = 0; i != number_of_items; i++)
329  sum += items[i];
330 
331  // A função retorna devolvendo a média pretendida, que é dada pelo
332  // quociente entre a soma calculada e o número de items somados.
333  return sum / number_of_items;

338 // O procedimento `main()`
339 // -----------------------
340 //
341 // Este procedimento é o ponto de entrada dos programas em C. É a primeira
342 // rotina a ser invocada nos programas em C. Neste caso particular queremos
343 // ignorar possíveis argumentos da linha de comandos, pelo que declaramos o
344 // procedimento colocando apenas `void` na lista de parâmetros.
345 //
346 // Note que, ao contrário do que acontece noutras linguagens, `int main()` não
347 // significa que o procedimento não receba qualquer argumento: significa que
348 // recebe argumentos não especificados. Trata-se de um resquício do passado da
349 // linguagem C que há que evitar a todo o custo. Para indicar que uma rotina não
350 // recebe qualquer argumento deve usar-se sempre `void` na lista de parâmetros.
351 //
352 // Apesar de se tratar de um procedimento, não se usa `void` como tipo de
353 // devolução. Isso deve-se ao facto de os possível erros ocorridos durante a
354 // execução do programa serem codificados através do valor inteiro devolvido.
355 // Esta prática é comum e foi apresentada mais atrás. O valor a devolver é
356 // usualmente 0, mas em rigor depende do sistema operativo (estamos a devolver
357 // um valor ao retornar da execução da primeira rotina do programa, pelo que o
358 // valor devolvido é usado pela entidade que executou o programa, possivelmente
359 // o sistema operativo). Assim, é comum usar as macros `EXIT_SUCCESS` para
360 // assinalar sucesso e `EXIT_FAILURE` para assinalar um erro.
361 //
362 // O programa, no nosso caso, tem um objectivo simples: escrever no ecrã uma
363 // sequência de idades de alunos dada e, em seguida, indicar qual a média de
364 // idades desses alunos.
365 int main(void)
366 {
367  // Definição do _array_ `student_ages` com inicialização explícita. Note
368  // que não indicámos explicitamente o tamanho do _array_, pelo que o
369  // compilador o calculará contando o número de valores da lista de
370  // inicializadores. Neste caso, por isso, o número de itens é 4.
371  int student_ages[] = {23, 24, 25, 26};
372  // O C não fornece nenhuma forma directa de obter o número de itens de
373  // um _array_. No entanto, quando acesso ao _array_ propriamente dito, e
374  // não a um ponteiro para o seu primeiro item, podemos usar um truque
375  // interessante para obter esse número: usar o operador `sizeof` para
376  // obter o espaço total em octetos ocupado pelo _array_ em memória e
377  // dividi-lo pelo espaço ocupado em memória por cada um dos seus itens.
378  // É isso que é feito neste local. O resultado é guardado numa
379  // _constante_, ou seja, numa «variável» que, devido ao qualificador
380  // `const`, o compilador não permitirá que seja alterada. Note que é boa
381  // ideia declarar explicitamente como constantes todos os _objectos_ que
382  // não se pretenda alterar depois de inicializados.
383  const int number_of_students =
384  sizeof(student_ages) / sizeof(int);
385 
386  // Invocação do procedimento `print()`, definido noutro local, passando
387  // o número de alunos como primeiro argumento e o _array_ com as suas
388  // idades como segundo argumento. Mais uma vez, é importante perceber
389  // que, na realidade, e ao contrário do que a frase anterior sugere, o
390  // que é passado como primeiro argumento _não é_ o _array_
391  // `student_ages`, mas sim _um ponteiro para o seu primeiro item_.
392  print(number_of_students, student_ages);
393 
394  // Impressão da média no ecrã. A média é obtida como resultado da
395  // invocação da função `average_of()` passando-lhe os parâmetros
396  // apropriados.
397  printf("The average age is %g.\n",
398  average_of(number_of_students, student_ages));
399 
400  // Retorno do procedimento devolvendo ao sistema operativo o valor
401  // apropriado para assinalar que a execução teve sucesso.
402  return EXIT_SUCCESS;
403 }
