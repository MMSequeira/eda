4 // O que são os _arrays_
5 // ---------------------
6 //
7 // Os _arrays_ são uma das formas de agregar dados em C. Nos _arrays_ os dados,
8 // a que chamamos «itens», são todos do mesmo tipo. Estes dados estão
9 // organizados numa sequência com um comprimento fixo. Por exemplo,
10 //
11 // ```C
12 // int array[4];
13 // ```
14 // define um novo _array_ com exactamente quatro inteiros.
15 
16 // Tal como acontece no caso de variáveis dos tipos básicos da linguagem, por
17 // omissão o C _não_ inicializa os itens de um _array_ (desde que seja um
18 // _array_ automático, pois se for estático todos os itens são inicializados com
19 // um valor nulo). Ou seja, estes começam a sua vida com valores que, do ponto
20 // de vista do programador, são puro lixo, mesmo que provavelmente estes valores
21 // consistam simplesmente numa interpretação do padrão de _bits_ que se encontra
22 // na memória alocada para esses itens ou variáveis.
23 //
24 // Os _arrays_ podem ser inicializados explicitamente através de uma lista
25 // de inicializadores:
26 //
27 // ```C
28 // int array[4] = {0, 1, 2};
29 // ```
30 //
31 // Neste caso se omitimos o valor inicial do quarto item do _array_. Quando isto
32 // acontece, o compilador do C encarrega-se de gerar código máquina que
33 // inicializa os itens em falta com valores nulos. Podemos, por isso, usar um
34 // truque simpático para inicializar todos os itens dos _arrays_ do C com o
35 // valor zero (no entanto, a norma que especifica o C11 não é clara quanto à
36 // validade de usar uma lista de inicialização vazia...):
37 //
38 // ```C
39 // int array[4] = {};
40 // ```
41 //
42 // Quando se fornece uma lista de inicialização, é possível omitir a
43 // dimensão dos _arrays_. Nesse caso, o _array_ terá como dimensão o número
44 // de valores colocado entre chavetas.

51 // Incluímos o ficheiro de cabeçalho `array_utils.h`, pois contém a declaração
52 // do procedimento `void print(int number_of_items, int
53 // items[number_of_items])`, usado mais abaixo. A inclusão está a ser feita
54 // encerrando o nome do ficheiro entre aspas, e não entre parênteses agudos.
55 // Isso faz com que o compilador procure o ficheiro não apenas nos locais padrão
56 // habituais (e.g., `/usr/include`) mas também no próprio projecto em que nos
57 // encontramos. Neste caso o ficheiro está no projecto Code::Blocks
58 // `array_utils`. Note que não basta que este ficheiro de cabeçalho, contendo o
59 // _protótipo_, esteja disponível para inclusão durante o processo de construção
60 // (mais precisamente durante o pré-processamento): é necessário que esteja
61 // também disponível uma biblioteca ou um ficheiro objecto com o resultado da
62 // compilação do código fonte correspondente. Essa biblioteca ou ficheiro
63 // objecto será _fundido_ com os restantes ficheiros objecto e bibliotecas
64 // durante o passo da _fusão_ do código.
65 #include "array_utils.h"
