1 // `print.c` &ndash; Ficheiro de implementação (modularização física)
2 // ==================================================================
3 //
4 // Estrutura elementar de um ficheiro de implementação
5 // ---------------------------------------------------
6 //
7 // Este é um dos ficheiros de implementação associados ao ficheiro de cabeçalho
8 // [`array_utils.h`](array_utils.html). Neste caso optou-se por manter um único
9 // ficheiro de cabeçalho e por dividir a implementação por vários ficheiros de
10 // implementação. Dessa forma:
11 //
12 // - Simplifica-se a vida de quem pretende utilizar os utilitários, pois terá de
13 // incluir apenas um ficheiro de cabeçalho. Claro está que esta vantagem se
14 // pode transformar numa desvantagem caso a quantidade de utilitários aumente
15 // substancialmente.
16 //
17 // - Facilita-se a manutenção das implementações, pois estas ficam separadas,
18 // cada uma em seu ficheiro.
19 //
20 // Note que as implementações dos vários utilitários são fornecidas na forma de
21 // uma _biblioteca_, ou seja, na forma de um ficheiro que arquiva os ficheiros
22 // objecto correspondentes a cada ficheiro de implementação. Isso significa que
23 // a organização das implementações apenas num grande ficheiro ou, como aqui se
24 // sugere, em ficheiros separados para cada utilitário, _é totalmente
25 // transparente_ para o programador cliente da biblioteca.
26 //
27 // Note ainda que, em Linux, a biblioteca terá o nome `libarray_utils.a`. Em C,
28 // o nome do ficheiro com a versão estática de uma biblioteca chamada
29 // <code>_biblioteca_</code> é usualmente <code>lib_biblioteca_.a</code>.
30 
31 // ### Inclusão do cabeçalho correspondente a esta implementação
32 //
33 // Antes de qualquer outra inclusão, incluímos o ficheiro de cabeçalho
34 // correspondente a este ficheiro de implementação. Isso ajuda a detectar
35 // inconsistências entre o ficheiro de implementação e o correspondente ficheiro
36 // de cabeçalho.
37 #include "array_utils.h"

43 //

51 // ### Definição de rotinas
52 //
53 // Definição de todas as rotinas correspondentes a este ficheiro. Neste caso há
54 // apenas uma rotina a definir: `print()`.
55 
56 // Definição da rotina `print()`. A definição das rotinas repete (porque o C o
57 // impõe) o seu cabeçalho, mas segue-o de um corpo onde se define o
58 // funcionamento ou implementação da rotina. Note que normalmente não se incluem
59 // comentários de documentação (e.g., [Doxygen](http://doxygen.org)) nos
60 // ficheiros de implementação.
61 void print(int number_of_items, int items[number_of_items])
62 {
63  // Verifica explicitamente as pré-condições desta rotina de que o
64  // primeiro argumento não pode ser negativo e o segundo não pode ser um
65  // ponteiro nulo.
66  assert(number_of_items >= 0);
67  assert(items != NULL);
68 
69  // Note a utilização de uma guarda fraca, aqui. Ou seja, usamos `i !=
70  // number_of_items` em vez da guarda mais usual (e mais forte) `i <
71  // number_of_items`. A vantagem de uma guarda fraca é que leva mais
72  // facilmente a ciclos infinitos (ou, pelo menos, muito longos) _quando
73  // o código está errado_. Isso tem a vantagem de facilitar a detecção de
74  // erros. Por exemplo, o que aconteceria se a asserção acima estivesse
75  // desactivada (usual no código em produção) e fosse passado um valor
76  // negativo como primeiro argumento? No caso da guarda forte, o ciclo
77  // terminaria e tudo _pareceria_ funcionar. No caso da guarda fraca, o
78  // ciclo tornar-se-ia muito longo (mas não infinito) o que levaria à
79  // mais rápida detecção do problema.
80  for (int i = 0; i != number_of_items; i++)
81  printf("%d\n", items[i]);
82 }
