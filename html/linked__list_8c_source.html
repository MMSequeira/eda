32 // ### Documentação da estrutura
33 //
34 // Como habitualmente, precedemos a definição da estrutura do correspondente
35 // comentário de documentação do [Doxygen](doxygen.org).
49 // ### Definição da estrutura
50 //
51 // As _estruturas C_ são um mecanismo da linguagem C que permite definir novos
52 // tipos de dados. As variáveis definidas como tendo como tipo uma estrutura C
53 // agregam um conjunto de membros ou campos de diferentes tipos. As estruturas C
54 // são uma das duas categorias de tipos ditos _tipos agregados_ do C, em
55 // conjunto com os _arrays_. As estruturas C são muitas vezes essenciais para
56 // criar _estruturas de dados_, mas não devem ser confundidas com estas. As
57 // estruturas C são um mecanismo básico da linguagem C, enquanto as estruturas
58 // de dados são formas de organização dos dados que transcendem a linguagem de
59 // programação em particular em que estão implementadas.
60 //
61 // As _cadeias simplesmente ligadas_ são também conhecidas por _listas
62 // simplesmente ligadas_, mas preferimos reservar a palavra «lista» para _tipos
63 // abstractos de dados_ (TAD), que podem ou não recorrer a cadeias simples ou
64 // duplamente ligadas como forma de implementação. As cadeias simplesmente
65 // ligadas consistem numa sequência de _elos_ (ou _nós_), cada um deles
66 // guardando dados como _carga útil_ e contendo um ponteiro para o elo seguinte
67 // na cadeia. Estas cadeias dizem-se simplesmente ligadas por cada um dos seus
68 // elos ter informação apenas acerca do _elo seguinte_ na cadeia. Quando os elos
69 // contêm também um ponteiro para o _elo anterior_, dizem-se _cadeias duplamente
70 // ligadas_.
71 //
72 // No exemplo abaixo recorre-se directamente à manipulação da estrutura C
73 // `struct singly_linked_int_list_node`, que nos permite implementar cadeias
74 // simplesmente ligadas, mas é raro que as estruturas de dados sejam usadas
75 // directamente no código _cliente_. As estruturas de dados são usadas sobretudo
76 // como implementação de tipos abstractos de dados. O mesmo deveria acontecer
77 // neste caso, mas optámos por deixar a estrutura de dados «nua» para se
78 // perceber melhor como os elos se organizam e como se acede e manipula uma
79 // cadeia simplesmente ligada.
80 //
81 // O nome da estrutura C definida, `struct singly_linked_int_list_node` reflecte
82 // os seguintes factos:
83 //
84 // - Trata-se de uma estrutura C: `struct` (esta palavra-chave faz parte do nome
85 // da estrutura, mas não da sua _etiqueta_, que consiste apenas no
86 // identificador `singly_linked_int_list_node`).
87 // - Trata-se de um _elo de uma cadeia_: `list_node`.
88 // - Trata-se de um elo de _uma cadeia_: `list`.
89 // - Trata-se de um elo de uma cadeia _simplesmente ligada_: `singly_linked`.
90 // - Trata-se de um elo _cuja carga útil é um `ìnt`_: `int`.
91 //
92 // As variáveis pertencentes ao tipo `struct singly_linked_int_list_node` são
93 // usualmente elos de cadeias simplesmente ligadas. Cada uma dessas variáveis
94 // contém dois membros:
95 //
96 // - `value` &ndash; Um campo do tipo `int` correspondendo à _carga útil_ do elo
97 // em causa.
98 // - `next_node` &ndash; Um _ponteiro para o elo seguinte_ da cadeia
99 // simplesmente ligada a que o elo pertence.
100 //
101 // Note que o campo `next_node` é definido recorrendo à própria estrutura C em
102 // definição. Esta definição recursiva da estrutura C é perfeitamente legítima,
103 // pois o campo está a ser definido como sendo um _ponteiro_ para uma instância,
104 // objecto ou variável do mesmo tipo.
105 struct singly_linked_int_list_node {
106  int value;
107  struct singly_linked_int_list_node *next_node;
108 };

110 // Criação e manipulação de uma cadeia simplesmente ligada
111 // -------------------------------------------------------
112 //
113 // Usamos a rotina `main()` para criar e manipular uma cadeia simplesmente
114 // ligada.
115 
116 // ### Documentação da rotina `main()`
117 //
118 // Como habitualmente, usamos o formato [Doxygen](doxygen.org) para os
119 // comentários de documentação.
127 // ### Definição da rotina `main()`
128 int main(void)
129 {
130  // Para podermos manipular uma cadeia simplesmente ligada, precisamos de
131  // uma forma de saber onde está o seu primeiro elo. Fazemo-lo através da
132  // variável `first_node`, que serve justamente para apontar o primeiro
133  // elo. Note que o nome desta variável é um pouco enganador: na
134  // realidade esta variável não guarda o primeiro elo, mas sim _um
135  // ponteiro para o primeiro elo_. Pode confirmar que se trata de um
136  // ponteiro através do símbolo `*` antes do seu nome. Os nomes dos
137  // restantes ponteiros usados neste programa seguem o mesmo padrão: o
138  // nome refere-se àquilo que é apontado, e não ao ponteiro propriamente
139  // dito.
140  //
141  // As cadeias ligadas só são verdadeiramente úteis se o programador
142  // tiver a possibilidade de criar novos elos e de os inserir na cadeia,
143  // ou de os remover e destruir, nos instantes em que isso lhe é mais
144  // conveniente. Por isso, os elos das cadeias ligadas são tipicamente
145  // _variáveis dinâmicas_, ou seja, variáveis construídas pelo código do
146  // programador na memória dinâmica por ele reservada usando a rotina
147  // `malloc()` (ou das outras rotinas do C com objectivo semelhante),
148  // numa zona da memória do programa a que se chama normalmente o
149  // «montão» ou _heap_, e destruídas pelo código do programador
150  // libertando essa mesma memória através do procedimento `free()`.
151  //
152  // Esta instrução define a variável local `first_node` como um ponteiro
153  // para `struct singly_linked_int_list_node` e inicializa esse ponteiro
154  // com o endereço de uma zona de memória com dimensão apropriada para
155  // guardar uma instância desse tipo. Essa zona de memória será
156  // inicializada nas instruções seguintes como sendo o primeiro e único
157  // elo da cadeia.
158  struct singly_linked_int_list_node *first_node =
159  malloc(sizeof(struct singly_linked_int_list_node));
160  // Para perceber o código de inicialização, bem como o que se segue,
161  // precisa de se recordar que o operador `->` serve para aceder a um
162  // campo de uma estrutura através de um seu ponteiro. Assim, o código
163  // apresentado é equivalente a:
164  //
165  // ```C
166  // (*first_node).value = 10;
167  // (*first_node).next_node = NULL;
168  // ```
169  //
170  // A utilização de parênteses no código acima deve-se à maior
171  // precedência do operador `.` relativamente ao operador `*` (ver
172  // [tabela de precedência e associatividade dos operadores do C e do
173  // C++ na Wikipédia](http://bit.ly/147TVC2)).
174  //
175  // Através do ponteiro `first_node`, inicializamos os dois membros ou
176  // campos do primeiro e único elo da cadeia. Esse elo contém como carga
177  // útil o inteiro 10. A inicialização tem de incidir também sobre o
178  // campo `next_node`, que de outro modo conteria um valor indefinido.
179  // Mas que endereço guardar nesse campo, um ponteiro para o elo seguinte
180  // na cadeia, se a cadeia ainda só tem um elo? A resposta é simples: o
181  // último elo da cadeia caracteriza-se por ter o valor especial `NULL`
182  // nesse campo, indicando-se com isso que não existe qualquer outro elo
183  // da cadeia depois dele.
184  first_node->value = 10;
185  first_node->next_node = NULL;
186 
187  // Vamos agora percorrer a cadeia simplesmente ligada desde o primeiro
188  // elo até o seu fim. Para isso usamos o ciclo `for` de um forma um
189  // pouco diferente do usual. Na inicialização do ciclo `for`, definimos
190  // uma nova variável `node`, um ponteiro para `struct
191  // singly_linked_int_list_node`, que inicializamos com o endereço do
192  // primeiro elo: `struct singly_linked_int_list_node *node =
193  // first_node`. No progresso do ciclo `for`, fazemos o ponteiro `node`
194  // avançar, atribuindo-lhe o endereço do elo seguinte em relação ao elo
195  // apontado pela própria variável `node`: `node = node->next_node`. Como
196  // guarda do ciclo, usamos o predicado `node != NULL`, ou seja, o ciclo
197  // continua, avançando com o ponteiro `node` ao longo dos elos da
198  // cadeia, até atingir o valor `NULL`, guardado no campo `next` do
199  // último elo da cadeia. Em cada passo do ciclo, escrevemos no ecrã o
200  // valor da carga útil do elo corrente da cadeia.
201  //
202  // Se lhe parece um pouco exagerado usar um ciclo `for` para percorrer
203  // uma cadeia que neste ponto tem apenas um elo, tem toda a razão. É um
204  // exagero. Mas pelo menos deixa claro que podemos usar um ciclo para
205  // percorrer cadeias com qualquer comprimento (inclusive com comprimento
206  // nulo, como veremos).
207  //
208  // Naturalmente, o resultado da execução deste código será:
209  //
210  // ```
211  // After the first node insertion:
212  // Node payload: 10
213  // ```
214  printf("After the first node insertion:\n");
215  for (struct singly_linked_int_list_node *node = first_node;
216  node != NULL;
217  node = node->next_node)
218  printf("Node payload: %d\n", node->value);
219  putchar('\n');
220 
221  // Vamos agora inserir um novo elo na cadeia, no seu final. Para isso,
222  // reservamos a quantidade de memória apropriada e colocamos o seu
223  // endereço no membro ponteiro `next_node` do primeiro elo da cadeia,
224  // apontado por `first_node`.
225  first_node->next_node =
226  malloc(sizeof(struct singly_linked_int_list_node));
227  // Da mesma forma que anteriormente, inicializamos os membros do novo
228  // elo com valores apropriados, nomeadamente com 20 na carga útil
229  // `value` e com o valor especial `NULL` no ponteiro `next_node`,
230  // deixando claro que não há qualquer elo após este novo elo, que é
231  // agora o último da cadeia.
232  first_node->next_node->value = 20;
233  first_node->next_node->next_node = NULL;
234 
235  // Mostramos no ecrã o conteúdo da cadeia, tal como anteriormente. O
236  // resultado da execução deste código será:
237  //
238  // ```
239  // After the second node insertion:
240  // Node payload: 10
241  // Node payload: 20
242  // ```
243  printf("After the second node insertion:\n");
244  for (struct singly_linked_int_list_node *node = first_node;
245  node != NULL;
246  node = node->next_node)
247  printf("Node payload: %d\n", node->value);
248  putchar('\n');
249 
250  // Desta vez vamos inserir um novo elo _no início_ da cadeia. Para isso,
251  // começamos por construir um novo elo, guardando o seu endereço numa
252  // variável local `new_node`.
253  struct singly_linked_int_list_node *new_node =
254  malloc(sizeof(struct singly_linked_int_list_node));
255  // Inicializamos a carga útil do novo elo com o valor `int` 0 (zero). O
256  // elo seguinte da cadeia, relativamente ao novo elo, é o elo que neste
257  // momento é o primeiro elo da cadeia. Isto acontece porque o novo elo
258  // está a ser inserido no início da cadeia, ao contrário do que fizemos
259  // antes.
260  new_node->value = 0;
261  new_node->next_node = first_node;
262  // Construído o novo elo, já com uma referência apropriada para o elo da
263  // cadeia que se lhe segue, precisamos de actualizar o ponteiro
264  // `first_node` de modo a apontar para o novo elo, que passará a ser o
265  // primeiro elo da cadeia.
266  first_node = new_node;
267 
268  // Mostramos no ecrã o conteúdo da cadeia, tal como anteriormente. O
269  // resultado da execução deste código será:
270  //
271  // ```
272  // After the third node insertion:
273  // Node payload: 0
274  // Node payload: 10
275  // Node payload: 20
276  // ```
277  printf("After the third node insertion:\n");
278  for (struct singly_linked_int_list_node *node = first_node;
279  node != NULL;
280  node = node->next_node)
281  printf("Node payload: %d\n", node->value);
282  putchar('\n');
283 
284  // Vamos agora remover o primeiro elo da cadeia. Para isso, poderia
285  // parecer suficiente fazer avançar o ponteiro `first_node` usando a
286  // atribuição `first_node = first_node->next_node`. Mas a verdade é que
287  // precisamos de libertar a memória ocupada pelo primeiro elo, pelo que
288  // temos de guardar o seu endereço numa variável auxiliar.
289  struct singly_linked_int_list_node *node_to_free = first_node;
290  first_node = first_node->next_node;
291  free(node_to_free);
292 
293  // Mostramos no ecrã o conteúdo da cadeia, tal como anteriormente. O
294  // resultado da execução deste código será:
295  //
296  // ```
297  // After the first node removal:
298  // Node payload: 10
299  // Node payload: 20
300  // ```
301  printf("After the first node removal:\n");
302  for (struct singly_linked_int_list_node *node = first_node;
303  node != NULL;
304  node = node->next_node)
305  printf("Node payload: %d\n", node->value);
306  putchar('\n');
307 
308  // Podemos também remover o último elo da cadeia, desde que tenhamos uma
309  // forma simples de aceder ao seu penúltimo elo. Neste momento o
310  // penúltimo elo é também o primeiro elo da cadeia, pelo que o acesso é
311  // fácil.
312  struct singly_linked_int_list_node *next_to_last_node = first_node;
313  // A remoção propriamente dita tem dois passos. O primeiro passo é
314  // libertar a memória reservada para o último elo.
315  free(next_to_last_node->next_node);
316  // O segundo passo é actualizar o campo `next_node` do que agora é o
317  // último elo, atribuindo-lhe o valor especial `NULL`. Isso marca-o como
318  // último elo da cadeia.
319  next_to_last_node->next_node = NULL;
320 
321  // Mostramos no ecrã o conteúdo da cadeia, tal como anteriormente. O
322  // resultado da execução deste código será:
323  //
324  // ```
325  // After the second node removal:
326  // Node payload: 10
327  // ```
328  printf("After the second node removal:\n");
329  for (struct singly_linked_int_list_node *node = first_node;
330  node != NULL;
331  node = node->next_node)
332  printf("Node payload: %d\n", node->value);
333  putchar('\n');
334 
335  // Neste momento a cadeia tem apenas um elo. Removê-lo é fácil, bastando
336  // libertar a sua memória e colocar o valor especial `NULL` no ponteiro
337  // `first_node`, deixando assim claro que a cadeia está agora vazia.
338  free(first_node);
339  first_node = NULL;
340 
341  // Mostramos no ecrã o conteúdo da cadeia, tal como anteriormente. O
342  // ciclo não sofre qualquer alteração, funcionando por isso
343  // correctamente, _mesmo para cadeias vazias_. O resultado da execução
344  // deste código será:
345  //
346  // ```
347  // After the third node removal:
348  // ```
349  printf("After the third node removal:\n");
350  for (struct singly_linked_int_list_node *node = first_node;
351  node != NULL;
352  node = node->next_node)
353  printf("Node payload: %d\n", node->value);
354 
355  // Terminamos assinalando sucesso.
356  return EXIT_SUCCESS;
357 }
