20 // Para podemos calcular a partir de _arrays_ de `double`, de uma só vez, todas
21 // as estatísticas em que estamos interessados, é conveniente usar uma estrutura
22 // para as representar.
23 struct double_statistics {
24  // A média e o desvio padrão.
25  double average;
26  double stddev;
27  // A mediana, o valor mínimo e o valor máximo.
28  double median;
29  double minimum;
30  double maximum;
31 };

33 // Rotina que cria um novo _array_ com `length` items `double`, todos por
34 // inicializar. Devolve um ponteiro para o novo _array_. Devolve `NULL` em caso
35 // de erro.
36 double *new_double_array_of(long length);
37 
38 // Rotina que redimensiona um _array_ de `double` `items` recebido como
39 // argumento de modo a passar a conter `new_length` itens. Devolve um ponteiro
40 // para o _array_ redimensionado. O ponteiro devolvido pode ser igual ao
41 // ponteiro recebido como argumento, mas o é necessariamente. Se o _array_
42 // crescer, os novos itens estarão por inicializar. Devolve `NULL` em caso de
43 // erro. O valor de `new_length` não pode ser negativo. O valor de `items` pode
44 // ser `NULL`. Nesse caso admite-se que o _array_ a dimensionar tem zero itens.
45 double *resize_double_array_to(double *items, long new_length);
46 
47 // Rotina que lê todos os `double` que conseguir a partir do ficheiro cujo nome
48 // é passado como argumento e os guarda num _array_. Devolve o _array_ com os
49 // valores lidos do ficheiro. Devolve `NULL` em caso de erro. Em caso de
50 // sucesso, altera o inteiro apontado por `length` para o número de `double`
51 // lidos. Os valores de `file_name` e de `length` não podem ser `NULL`.
52 double *read_double_array_from(const char *file_name, long *length);
53 
54 // Procedimento que copia todos os `length` itens do _array_ de `double`
55 // `original` para o _array_ de `double` `copy`. Ambos os _arrays_ têm de conter
56 // pelo menos `length` items. O valor de `length` tem de ser não negativo. Os
57 // valores de `original` e de `copy` podem ser nulos, mas apenas se `length`
58 // tiver o valor zero.
59 void copy_double_array(long length, double copy[length],
60  const double original[length]);
61 
62 // Predicado que devolve `true` se os primeiros _length_ itens dos _arrays_ de
63 // `double` `first` e `second` forem iguais. Devolve `false` no caso contrário.
64 // O valor de `length` não pode ser negativo. Os valores de `first` e `second`
65 // podem ser `NULL`, mas apenas se o valor de `length` for zero.
66 bool double_arrays_equal(long length, const double first[length],
67  const double second[length]);

69 // Função que devolve o valor médio dos primeiros `length` itens do _array_ de
70 // `double` `items`. O valor de `length` não pode ser negativo. Devolve o valor
71 // especial NaN se `length` for zero. O valor de `items` pode ser `NULL`, mas

75 // Função que devolve o desvio padrão dos primeiros `length` itens do _array_ de
76 // `double` `items`. O valor de `length` não pode ser negativo. Devolve o valor
77 // especial NaN se `length` for zero. O valor de `items` pode ser `NULL`, mas
78 // apenas se `length` for zero.
79 double double_array_stddev(long length, const double items[length]);

81 // Função que devolve o valor mediano dos primeiros `length` itens do _array_ de
82 // `double` `items`. O valor de `length` não pode ser negativo. Devolve o valor
83 // especial NaN se `length` for zero. O valor de `items` pode ser `NULL`, mas
84 // apenas se `length` for zero.
85 double double_array_median(long length, const double items[length]);
86 
87 // Função que devolve o valor mínimo dos primeiros `length` itens do _array_ de
88 // `double` `items`. O valor de `length` não pode ser negativo. Devolve o valor
89 // ∞ se `length` for zero. O valor de `items` pode ser `NULL`, mas
90 // apenas se `length` for zero.
91 double double_array_minimum(long length, const double items[length]);
92 
93 // Função que devolve o valor máximo dos primeiros `length` itens do _array_ de
94 // `double` `items`. O valor de `length` não pode ser negativo. Devolve o valor
95 // -∞ se `length` for zero. O valor de `items` pode ser `NULL`, mas
96 // apenas se `length` for zero.
97 double double_array_maximum(long length, const double items[length]);
98 
99 // Função que devolve uma instância da estrutura `struct double_statistics`
100 // contendo várias estatísticas dos primeiros `length` itens do _array_ de
101 // `double` `items`. O valor de `length` não pode ser negativo. Se `length` for
102 // zero, a média, o desvio padrão e a mediana têm o valor especial NaN, o mínimo
103 // tem o valor ∞ e o máximo tem o valor -∞. O valor de `items` pode ser
104 // `NULL`, mas apenas se `length` for zero.
105 struct double_statistics double_array_statistics(long length,
