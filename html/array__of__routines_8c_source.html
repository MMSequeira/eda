array_of_routines.c
1 // `array_of_routines.c` &ndash; Ponteiros para rotinas e _arrays_
2 // ===============================================================
3 //
4 // Este programa ilustra uma forma particular de usar [ponteiros para
5 // rotinas](routine_pointers.c.html): guardando-os dentro de um _array_. Para
6 // informação especificamente sobre ponteiros para rotinas, sua utilização e
7 // particularidades da sua implementação na linguagem C, ver
8 // [routine_pointers.c](routine_pointers.c.html).
9 //
10 // Optámos por omitir os comentários de documentação do [Doxygen](doxygen.org),
11 // ao contrário do que tem sido nosso hábito.
12 
13 // Inclusões
14 // ---------
15 //
16 // As inclusões usuais de ficheiros de cabeçalho, para poder usar o procedimento
17 // `printf` de escrita no ecrã e para poder devolver `EXIT_SUCCESS` no final do
18 // programa.
19 #include <stdio.h>
20 #include <stdlib.h>

22 // Definição das rotinas a guardar no _array_
23 // ------------------------------------------
24 //
25 // Todos os itens de um _array_ em C têm de ser do mesmo tipo. Neste caso o
26 // _array_ guardará ponteiros para rotinas, que pode isso precisam de ter todos
27 // o mesmo tipo. Ora, o tipo de um ponteiro para um rotina inclui a _assinatura_
28 // completa da rotina, com excepção dos nomes da rotina e dos seus parâmetros.
29 // Ou seja, inclui:
30 //
31 // - A sequência dos tipos dos parâmetros da rotina.
32 // - O tipo de devolução da rotina.
33 //
34 // No nosso caso, todas as rotinas serão funções com dois parâmetros do tipo
35 // `int` e devolvendo também um `int`. Tratando-se de um mero exemplo, optámos
36 // por guardar no _array_ apenas quatro funções, correspondendo às quatro
37 // operações aritméticas básicas.
38 
39 // ### Definição da função `sum()`
40 //
41 // Devolve a soma dos seus dois argumentos.
42 int sum(int first, int second)
43 {
44  return first + second;
45 }
46 
47 // ### Definição da função `subtraction()`
48 //
49 // Devolve a diferença entre os seus dois argumentos.
50 int subtraction(int first, int second)
51 {
52  return first - second;
53 }
54 
55 // ### Definição da função `product()`
56 //
57 // Devolve o produto dos seus dois argumentos.
58 int product(int first, int second)
59 {
60  return first * second;
61 }
62 
63 // ### Definição da função `product()`
64 //
65 // Devolve o quociente do seu primeiro argumento quando dividido pelo seu
66 // segundo argumento.
67 int division(int first, int second)
68 {
69  return first / second;
70 }

72 // Definição do _array_ de operações
73 // ---------------------------------
74 //
75 // O _array_ será definido de forma a ser constante e global. A opção pela
76 // constância do _array_ prende-se com o seu carácter global. É que, se a
77 // utilização de variáveis globais é geralmente um má prática, por introduzir
78 // formas subtis e pouco evidentes de comunicação entre troços distantes de um
79 // programa, já a utilização de constantes globais não sofre do mesmo problema,
80 // sendo por isso a sua utilização aconselhável em muitas circunstâncias.
81 //
82 // A definição do _array_ de ponteiros para as funções definidas é difícil de
83 // perceber, devido à sintaxe peculiar do C. Há, no entanto, uma forma de
84 // interpretação que torna a tarefa de compreender esta definição bem mais
85 // simples: a definição de uma entidade em C reflecte a sua utilização no resto
86 // do código. Por exemplo, se definirmos uma variável `an_int` como sendo do
87 // tipo `int`, então poderemos usá-la no código, por exemplo, numa atribuição a
88 // uma outra variável do mesmo tipo, da seguinte forma:
89 //
90 // ```C
91 // // Definition of another_int and an_int...
92 // ...
93 // // Use of the variables:
94 // another_int = an_int;
95 // ```
96 //
97 // Ou seja, escrevendo `an_int` obtemos num `int`. A definição de `an_int`
98 // reflecte esse facto: começa por se dizer o tipo resultante, neste caso `int`,
99 // escreve-se depois a expressão que resulta nesse tipo, neste caso simplesmente
100 // `an_int`, e finalmente termina-se a instrução de definição com o terminador
101 // `;`:
102 //
103 // ```C
104 // int an_int;
105 // ```
106 //
107 // ou, procedendo a uma inicialização durante a definição da variável,
108 //
109 // ```C
110 // int an_int = 3;
111 // an_int = 0; // valid: assignment changing the the value of the variable.
112 // ```
113 //
114 // Se em vez de uma variável pretendêssemos uma constante `a_constant_int`,
115 // bastaria dizer que a expressão de `a_constant_int` resultaria num inteiro
116 // constante:
117 //
118 // ```C
119 // int const a_constant_int = 3;
120 // a_constant_int = 0; // invalid: cannot change a constant!
121 // ```
122 //
123 // O C permite trocar a posição do qualificador `const`, que se pode colocar,
124 // por isso, na sua posição mais habitual:
125 //
126 // ```C
127 // const int a_constant_int = 3;
128 // a_constant_int = 0; // invalid: cannot change a constant!
129 // ```
130 //
131 // Suponha-se agora uma variável `pointer_to_an_int` que é um ponteiro para
132 // `int`, ou seja, que guarda o endereço de um `int`. Assumindo que a variável
133 // contém um endereço válido, como se pode aceder ao inteiro por ela apontado?
134 // Fácil: usando o operador conteúdo, ou de «desreferenciação», `*`. Por
135 // exemplo:
136 //
137 // ```C
138 // // Definition and initialization of an_int:
139 // int an_int = 3;
140 // // Definition of pointer_to_an_int:
141 // ...
142 // // Assignment of the address of an_int to the pointer pointer_to_an_int:
143 // pointer_to_an_int = &an_int;
144 // // Definition of another_int:
145 // int another_int;
146 // // Assignment of the contents of the integer pointed by pointer_to_an_int
147 // // to variable another_int:
148 // another_int = *pointer_to_an_int;
149 // ```
150 //
151 // Ou seja, escrevendo `*pointer_to_an_int` obtemos um `int`. A definição de
152 // `pointer_to_an_int` reflecte esse facto: começa por se dizer o tipo
153 // resultante, neste caso `int`, escreve-se depois a expressão que resulta nesse
154 // tipo, neste caso `*pointer_to_an_int`, e finalmente termina-se a instrução
155 // de definição com o terminador `;`:
156 //
157 // ```C
158 // int *pointer_to_an_int;
159 // ```
160 //
161 // ou, procedendo a uma inicialização durante a definição da variável,
162 //
163 // ```C
164 // int *pointer_to_an_int = &an_int;
165 // pointer_to_an_int = &some_other_int; // valid: assignment changes pointer.
166 // ```
167 //
168 // Se em vez de uma variável pretendêssemos uma constante
169 // `constant_pointer_to_an_int`, bastaria dizer que a expressão
170 // `constant_pointer_to_an_int` resultaria num ponteiro constante para `int`:
171 //
172 // ```C
173 // int *const constant_pointer_to_an_int = &an_int;
174 // constant_pointer_to_an_int =
175 // &some_other_int; // invalid: cannot change the constant pointer!
176 // ```
177 //
178 // Estranho? Nem por isso. Como queremos que `constant_pointer_to_an_int` seja
179 // um ponteiro constante para `int`, temos de colocar o qualificador após o `*
180 // `. Se colocássemos o qualificador `const` antes do `*` estaríamos a dizer que
181 // a expressão `*constant_pointer_to_an_int` tinha o tipo `int const` (ou `const
182 // int`, que é o mesmo). Ou seja, estaríamos a dizer que o ponteiro em si era
183 // variável, mas apontava para `int` constantes, que é como quem diz que os
184 // `int` apontados não poderiam ser alterados através do ponteiro. Vamos
185 // experimentar? Aqui vai:
186 //
187 // ```C
188 // const int *pointer_to_a_constant_int = &an_int;
189 // *pointer_to_a_constant_int =
190 // 0; // invalid: cannot change pointed constant int.
191 // pointer_to_a_constant_int =
192 // &some_other_int; // valid: assignment changes pointer.
193 // ```
194 //
195 // Podemos mesmo combinar as duas formas e definir um ponteiro constante para um
196 // `int` constante:
197 //
198 // ```C
199 // const int *const constant_pointer_to_a_constant_int = &an_int;
200 // *constant_pointer_to_a_constant_int =
201 // 0; // invalid: cannot change pointed constant int.
202 // constant_pointer_to_a_constant_int =
203 // &some_other_int; // invalid: cannot change the constant pointer!
204 // ```
205 //
206 // Suponha-se agora que pretendemos definir um _array_ de 10 `int` chamado
207 // `ten_ints`. Depois de o definir, podemos usá-lo em expressões de indexação,
208 // obtendo-se inteiros no processo:
209 //
210 // ```C
211 // // Definition of ten_ints with values 20 down to 11:
212 // ...
213 // // Definition of an_int:
214 // int an_int;
215 // // Assignment of the value of the first int in array ten_ints to variable
216 // // an_int:
217 // an_int = ten_ints[0];
218 // ```
219 //
220 // Ou seja, escrevendo `ten_ints[`_n_`]`, com _n_ entre 0 e 9, obtemos um `int`.
221 // A definição de `ten_ints` reflecte esse facto: começa por se dizer o tipo
222 // resultante, neste caso `int`, escreve-se depois a expressão que resulta nesse
223 // tipo, neste caso `ten_ints[10]`, e finalmente termina-se a instrução de
224 // definição com o terminador `;`. Tratando-se da definição de um _array_, o que
225 // colocamos dentro dos parênteses rectos é o número de itens do _array_, ou
226 // seja, o seu comprimento:
227 //
228 // ```C
229 // int ten_ints[10];
230 // ```
231 //
232 // ou, procedendo a uma inicialização durante a definição do _array_,
233 //
234 // ```C
235 // int ten_ints[10] = {
236 // 20, 19, 18, 17, 16, 15, 14, 13, 12, 11, 11
237 // };
238 // ten_ints[0] = 0; // valid: assignment changes first item.
239 // ```
240 //
241 // Usando-se um inicializador, podemos deixar a tarefa de contar o número de
242 // itens ao compilador. Assim, poderíamos ter definido o _array_ como se segue:
243 //
244 // ```C
245 // int ten_ints[] = {
246 // 20, 19, 18, 17, 16, 15, 14, 13, 12, 11, 11
247 // };
248 // ```
249 //
250 // Se se pretender que os itens de um _array_ `ten_constant_ints` sejam
251 // constantes, ou seja, que as expressões `ten_constant_ints[`_n_`]` sejam do
252 // tipo `const int`, então temos de definir o _array_ como se segue:
253 // ```C
254 // const int ten_constant_ints[10] = {
255 // 20, 19, 18, 17, 16, 15, 14, 13, 12, 11, 11
256 // };
257 // ten_constant_ints[0] = 0; // invalid: cannot change the array items!
258 // ```
259 //
260 // Finalmente, suponhamos uma variável `pointer_to_funtion` que guarda o
261 // endereço de uma função com dois parâmetros do tipo `int` e que devolve também
262 // um `int`. Para invocar a função apontada por esta variável com os argumentos
263 // 2 e 3, por exemplo, temos de (a) desreferenciar o ponteiro usando o operador
264 // `*` e (b) invocar a função resultante usando o operador de invocação, que
265 // consiste num par de parênteses curvos `()` contendo a lista dos argumentos
266 // separados por vírgulas. Como o operador de invocação `()` tem precedência
267 // relativamente ao operador de desreferenciação `*` (ver [tabela de precedência
268 // e associatividade dos operadores do C e do C++ na
269 // Wikipédia](http://bit.ly/147TVC2)), temos de garantir que este último é
270 // aplicado ao ponteiro para a função antes do operador de invocação. Ou seja,
271 // temos de escrever `(*pointer_to_function)(2, 3)`. O resultado desta invocação
272 // é um `int`, uma vez que dissemos que a função tem dois parâmetros `int` e
273 // devolve também um `int`. Por exemplo:
274 //
275 //
276 // ```C
277 // // Definition of the sum() function:
278 // int sum(int first, int second)
279 // {
280 // return first + second;
281 // }
282 //
283 // // Definition of the subtraction() function:
284 // int subtraction(int first, int second)
285 // {
286 // return first - second;
287 // }
288 //
289 // // Definition of pointer_to_function pointing to the sum() function:
290 // ...
291 // // Definition of an_int:
292 // int an_int;
293 // // Assignment of the result of the call to the sum function with arguments
294 // // 2 and 3 through the pointer_to_function variable:
295 // an_int = (*pointer_to_function)(2, 3);
296 // ```
297 //
298 // Ou seja, escrevendo `(*pointer_to_function)(`_n_`, `_m_`)`, com _n_ e _m_
299 // valores inteiros, obtemos um `int`. A definição de `pointer_to_function`
300 // reflecte esse facto: começa por se dizer o tipo resultante, neste caso `int`,
301 // escreve-se depois a expressão que resulta nesse tipo, neste caso
302 // `(*pointer_to_function)(int, int)`, e finalmente termina-se a instrução de
303 // definição com o terminador `;`. Tratando-se da definição de um ponteiro para
304 // uma rotina, o que colocamos dentro dos parênteses curvos no local dos
305 // argumentos é o tipo de cada um dos parâmetros da rotina:
306 //
307 // ```C
308 // int (*pointer_to_function)(int, int);
309 // ```
310 //
311 // ou, procedendo a uma inicialização durante a definição do ponteiro,
312 //
313 // ```C
314 // int (*pointer_to_function)(int, int) = &sum;
315 // pointer_to_function = &subtraction; // valid: assignment changes pointer.
316 // *pointer_to_function = ...; // invalid: routines are unchangeable.
317 // ```
318 //
319 // O C é simpático, assumindo que qualquer referência ao nome de uma rotina na
320 // realidade se refere ao endereço dessa rotina. Assim, o código acima pode ser
321 // simplificado:
322 //
323 // ```C
324 // int (*pointer_to_function)(int, int) = sum;
325 // pointer_to_function = subtraction; // valid: assignment changes pointer.
326 // ```
327 //
328 // Se pretendermos definir uma constante `constant_pointer_to_function`, então
329 // teremos de preceder o nome da constante do qualificador `const`, da mesma
330 // forma que anteriormente:
331 //
332 // ```C
333 // int (*const constant_pointer_to_function)(int, int) = sum;
334 // pointer_to_function =
335 // subtraction; // invalid: cannot change a constant pointer.
336 // ```
337 //
338 // Estamos finalmente em condições de perceber como definir o nosso _array_ de
339 // quatro ponteiros _constantes_ para funções com dois parâmetros `int` e
340 // devolvendo um `int`. Suponhamos que este _array_ se chama `operators` e que o
341 // seu primeiro item é o ponteiro (constante) para a função `sum()`. Então as
342 // seguintes expressões têm os significados indicados:
343 //
344 // - `operators` - O _array_ completo (que o C, salvo raras excepções, considera
345 // como equivalente a um ponteiro para o seu primeiro item).
346 //
347 // - `operators[0]` - O primeiro item do _array_, que é um ponteiro para a
348 // função `sum()`.
349 //
350 // - `*operators[0]` - O conteúdo da posição de memória apontada pelo primeiro
351 // item do _array_, ou seja, a função `sum()` propriamente dita.
352 //
353 // - `(*operators[0])(2, 3)` - O resultado da invocação da função `sum()`
354 // apontada pelo primeiro item do _array_ passando-lhe os argumentos 2 e 3, ou
355 // seja, o valor 5. Note-se que a colocação dos parênteses curvos em torno de
356 // `*operators[0]` é necessária porque o operador de invocação `()`, usado
357 // neste caso com os operandos 2 e 3, _tem maior precedência que o operador
358 // `*`_ (ver [tabela de precedência e associatividade dos operadores do C e do
359 // C++ na Wikipédia](http://bit.ly/147TVC2)).
360 //
361 // Ou seja, escrevendo `(*operators[`_i_`])(`_n_`, `_m_`)`, com _i_ entre 0 e 3
362 // e com _n_ e _m_ valores inteiros, obtemos um `int`. A definição de
363 // `operators` reflecte esse facto: começa por se dizer o tipo resultante, neste
364 // caso `int`, escreve-se depois a expressão que resulta nesse tipo, neste caso
365 // `(*operators[4])(int, int)`, e finalmente termina-se a instrução de definição
366 // com o terminador `;`. Tratando-se da definição de um _array_, colocamos entre
367 // parênteses rectos o número de itens do _array_, neste caso 4. Tratando-se de
368 // um _array_ de ponteiros para rotinas, o que colocamos dentro dos parênteses
369 // curvos no local dos argumentos é o tipo de cada um dos parâmetros da rotina,
370 // neste caso dois `int`:
371 //
372 // ```C
373 // int (*operations[4])(int, int);
374 // ```
375 //
376 // Uma vez que desejamos que os itens do _array_ (os ponteiros) sejam
377 // constantes, colocamos o qualificador `const` entre o `*` e o nome do _array_:
378 //
379 // ```C
380 // int (*const operations[4])(int, int);
381 // ```
382 //
383 // Podemos inicializar o _array_ usando um inicializador com os endereços das
384 // várias funções:
385 //
386 // ```C
387 // int (*const operations[4])(int, int) = {
388 // &sum,
389 // &subtraction,
390 // &product,
391 // &division
392 // };
393 // ```
394 //
395 // Podemos omitir a indicação explícita do número de itens do _array_, deixando
396 // ao compilador o trabalho de calcular o número de itens a partir do número de
397 // itens no inicializador:
398 //
399 // ```C
400 // int (*const operations[])(int, int) = {
401 // &sum,
402 // &subtraction,
403 // &product,
404 // &division
405 // };
406 // ```
407 //
408 // Finalmente, tirando partido do facto de, salvo raras excepções, o C converter
409 // qualquer referência ao nome de um rotina num ponteiro para essa mesma rotina,
410 // podemos simplificar a definição anterior para a nossa versão definitiva:
411 //
412 // ```C
413 // int (*const operations[])(int, int) = {
414 // sum,
415 // subtraction,
416 // product,
417 // division
418 // };
419 // ```
420 int (*const operations[])(int, int) = {
421  sum,
422  subtraction,
423  product,
424  division
425 };
426 
427 // Usamos agora o truque usual para obter o número de itens de um _array_:
428 // dividir o espaço de memória total ocupado pelo _array_ pelo espaço ocupado
429 // por cada um dos seus itens. Para isso usamos o operador `sizeof`.
430 const int number_of_operations =
431  sizeof(operations) /
432  sizeof(operations[0]);
433 
434 // Definimos a rotina principal, onde experimentamos o conceito de _array_ de
435 // rotinas.
436 int main(void)
437 {
438  // Percorremos cada um dos itens do _array_ `operations`.
439  for (int i = 0; i != number_of_operations; i++)
440  // Invocamos com os argumentos 5 e 2 a função apontada por cada
441  // item do _array_. O resultado será, naturalmente:
442  //
443  // ```
444  // Operation 0: 7
445  // Operation 1: 3
446  // Operation 2: 10
447  // Operation 3: 2
448  // ```
449  //
450  // Note que, para invocar a função apontada pelo item `i` do
451  // _array_ `operations`, poderia parecer obrigatório usar o
452  // operador `*`:
453  //
454  // ```C
455  // printf("Operation %d: %d\n", i,
456  // (*operations[i])(5, 2));
457  // ```
458  //
459  // A linguagem C, no entanto, permite-nos omitir o operador `*`
460  // e aplicar directamente o operador de invocação `()` ao
461  // ponteiro para a função. Daí que o código apresentado seja
462  //
463  // ```C
464  // printf("Operation %d: %d\n", i,
465  // operations[i](5, 2));
466 
467  // ```
468  printf("Operation %d: %d\n", i,
469  operations[i](5, 2));
470 
471  // Terminamos assinalando sucesso.
472  return EXIT_SUCCESS;
473 }
