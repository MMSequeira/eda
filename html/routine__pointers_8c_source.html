routine_pointers.c
1 // `routine_pointers.c` &ndash; Ponteiros para rotinas
2 // ===================================================
3 //
4 // Um dos mecanismos que mais flexibilidade confere a uma linguagem de
5 // programação é a possibilidade de parametrizar módulos (e.g., rotinas ou, nas
6 // linguagens orientadas por objectos, classes ou objectos) com mecanismos. Ou
7 // seja, a possibilidade de deixar em aberto, durante a criação desses módulos,
8 // algumas das rotinas a que o módulo recorrerá. A indicação dessas rotinas é
9 // deixada a cargo do código cliente, i.e., do código que parametriza e usa
10 // esses módulos. Em C, o mecanismo que suporta este tipo de prática é o dos
11 // [ponteiros para rotinas](http://en.wikipedia.org/wiki/Function_pointer).
12 //
13 // Este programa exemplifica a utilização de ponteiros para rotinas em C,
14 // demonstrando duas formas de utilização em tudo equivalente. Uma dessas
15 // versões é mais «limpa», embora menos explícita. A outra é mais «suja» (no
16 // sentido em que o código é mais denso e difícil de ler), embora totalmente
17 // explícita. Em ambos os casos, o objectivo é parametrizar um procedimento de
18 // teste, passando-lhe como argumento uma de duas funções a testar. O programa
19 // exemplifica também a utilização de variáveis para guardar ponteiros para
20 // rotinas.
21

59 // Versão mais «limpa» e menos explícita do código
60 // -----------------------------------------------
61 //
62 // ### O procedimento `try_function()`
63 //
64 // O procedimento `try_function()` é um módulo (as rotinas são módulos)
65 // parametrizável por uma função. Essa função é recebida como argumento, sendo
66 // conhecida internamente ao procedimento por `f`. Note-se que o tipo de `f` é
67 // `int f(int)` (ver mais abaixo porque é que esta descrição do tipo de `f` está
68 // errada, na realidade), ou seja, uma função que recebe um inteiro como
69 // argumento e devolve um inteiro ao retornar. Este tipo é compatível com as
70 // definições das funções `double_of()` e `triple_of()`, que por isso podem ser
71 // passadas como argumento a este procedimento, como de resto acontece mais
72 // abaixo, no procedimento `main()`. O segundo parâmetro do procedimento
73 // `try_function()` guarda o valor a passar como argumento à função recebida
74 // como primeiro argumento quando esta for invocada.
75 void try_function(int f(int), int value)
76 {
77  // A invocação da função passada como argumento faz-se da forma usual,
78  // embora através de `f`. Recorde-se que `f` tanto pode ser a função
79  // `double_of` como a função `triple_pf`. Aliás, pode ser qualquer
80  // função com apenas um parâmetro `int` e que, ao terminar, devolva
81  // também um `int`. Neste local não precisamos de saber que função é:
82  // limitamo-nos a invocá-la.
83  printf("f(%d) = %d\n", value, f(value));
84 }
85 
86 //
87 int main()
88 {
89  // ### Invocação do procedimento definido
90  //
91  // Invocamos o procedimento `try_function()` definido acima, passando-
92  // lhe como primeiro argumento a função `double_of` e, depois, a função
93  // `triple_of`. Em ambos os casos o segundo argumento tem o valor 4. Os
94  // valores impressos no ecrã são, como seria de esperar, 8 e 12, ou
95  // seja, o dobro e o triplo de 4, respectivamente.
96  try_function(double_of, 4); // => f(4) = 8
97  try_function(triple_of, 4); // => f(4) = 12
98 
99  // ### Variáveis locais (apontando) rotinas
100  //
101  // Da mesma forma que podemos passar rotinas como argumento a outras
102  // rotinas, podemos guardar rotinas em variáveis. OK, OK. É falso. Não
103  // podemos. Temos vindo a descrever as coisas de forma simplista. Mas
104  // pelo menos avisámos! E neste ponto, já não nos podemos iludir mais: o
105  // que se pode guardar em variáveis são _ponteiros_ para rotinas. A
106  // linguagem C é que é simpática e nos poupa ao trabalho de indicar qual
107  // o verdadeiro tipo de uma rotina recebida como argumento. Tal como no
108  // caso dos _arrays_, o C transforma uma declaração de um rotina como
109  // parâmetro numa declaração de um _ponteiro_ para uma rotina. Mais
110  // sobre o assunto mais abaixo, na versão menos «limpa do código». Mas
111  // não o faz quando se trata de definir uma variável local. É por isso
112  // que a linha de código
113  //
114  // ```C
115  // int f(int) = double_of;
116  // ```
117  //
118  // produz um erro de compilação. Sem a indicação explícita de que se
119  // trata de um ponteiro, o C assume tratar-se da declaração (da
120  // existência algures) de uma função `f()`. O problema resolve-se
121  // indicando explicitamente que a variável `f` é um ponteiro para uma
122  // função. Note-se que o asterisco antes do nome da variável reflecte a
123  // forma como a variável será utilizada. Aliás, as declarações em C
124  // seguem esse padrão: indicar a forma de utilização da variável e
125  // indicar o tipo que resulta dessa utilização. Os parênteses em torno
126  // de `*f` devem-se ao facto de o operador de invocação de rotinas (os
127  // parênteses que se seguem ao nome da rotina e que envolvem os
128  // argumentos a passar a essa rotina) ter uma precedência superior à do
129  // operador de desreferenciação. Assim, `int *f(int)` seria interpretado
130  // como a declaração de existência algures de uma função `f()`,
131  // invocável com um único valor inteiro, e devolvendo, quando retorna,
132  // um _ponteiro_ para um valor inteiro.
133  int (*f)(int) = double_of;
134 
135  // Naturalmente, o resultado será a impressão no ecrã do dobro de cinco.
136  // Repare-se na forma simples de invocação.
137  printf("f(5) = %d\n", f(5)); // => f(5) = 10
138 
139  //
140  return EXIT_SUCCESS;
141 }

148 // Versão menos «limpa» e mais explícita do código
149 // -----------------------------------------------
150 //
151 // ### O procedimento `try_function()`
152 //
153 // O procedimento `try_function()` é um módulo (as rotinas são módulos)
154 // parametrizável por um ponteiro para uma função. Esse ponteiro para uma função
155 // é recebido como argumento, sendo conhecido internamente ao procedimento por
156 // `f`. Note-se que o tipo de `f` é `int (*f)(int)`, ou seja, um ponteiro para
157 // uma função que recebe um inteiro como argumento e devolve um inteiro ao
158 // retornar. O tipo deste ponteiro é compatível com as definições das funções
159 // `double_of()` e `triple_of()`, cujos endereços por isso podem ser passados
160 // como argumento a este procedimento, como de resto acontece mais abaixo, no
161 // procedimento `main()`. O segundo parâmetro do procedimento `try_function()`
162 // guarda o valor a passar como argumento à função recebida como primeiro
163 // argumento quando esta for invocada.
164 void try_function(int (*f)(int), int value)
165 {
166  // A invocação da função cujo endereço foi passado como argumento faz-se
167  // da forma usual, embora através de `f`. Recorde-se que `f` tanto pode
168  // apontar para a função `double_of` como para a função `triple_pf`.
169  // Aliás, pode apontar para qualquer função com apenas um parâmetro
170  // `int` e que, ao terminar, devolva também um `int`. Neste local não
171  // precisamos de saber que função é: limitamo-nos a invocá-la. A
172  // invocação é feita depois de desreferenciar o ponteiro usando o
173  // operador `*`.
174  printf("(*f)(%d) = %d\n", value, (*f)(value));
175 }
176 
177 //
178 int main()
179 {
180  // ### Invocação do procedimento definido
181  //
182  // Invocamos o procedimento `try_function()` definido acima, passando-
183  // lhe como primeiro argumento um ponteiro para a função `double_of` e,
184  // depois, um ponteiro para a função `triple_of`. Em ambos os casos o
185  // segundo argumento tem o valor 4. Note-se que se usa o operador
186  // endereço, ou seja, o operador `&` para obter os ponteiros a passar
187  // como argumento ao procedimento invocado. Os valores impressos no ecrã
188  // são, como seria de esperar, 8 e 12, ou seja, o dobro e o triplo de 4,
189  // respectivamente.
190  try_function(&double_of, 4); // => (*f)(4) = 8
191  try_function(&triple_of, 4); // => (*f)(4) = 12
192 
193  // ### Variáveis locais (apontando) rotinas
194  //
195  // Da mesma forma que podemos passar ponteiros para rotinas como argumento a outras
196  // rotinas, podemos guardar ponteiros para rotinas em variáveis. Os parênteses em torno
197  // de `*f` devem-se ao facto de o operador de invocação de rotinas (os
198  // parênteses que se seguem ao nome da rotina e que envolvem os
199  // argumentos a passar a essa rotina) ter uma precedência superior à do
200  // operador de desreferenciação.
201  int (*f)(int) = &double_of;
202 
203  // A invocação da função apontada por `f` é feita da forma usual, embora
204  // apenas depois de desreferenciar o ponteiro usando o operador
205  // conteúdo, ou seja, o operador `*`.
206  printf("(*f)(5) = %d\n", (*f)(5)); // => (*f)(5) = 10
207

209  return EXIT_SUCCESS;
210 }

214 // As regras
215 // ---------
216 //
217 // De forma resumida, as regras usadas pela linguagem C no que respeita a
218 // ponteiros para rotinas são as seguintes:
219 //
220 // - Sempre que um parâmetro for declarado como sendo uma rotina, o compilador
221 // interpreta-a como declarando um _ponteiro_ para rotinas do tipo indicado.
222 // Ou seja, o compilador transforma declarações de parâmetros da
223 // forma<code>_tipo_ _nome_(_tipos_dos_parânetros_)</code> em declarações da
224 // forma <code>_tipo_ (*_nome_)(_tipos_dos_parânetros_)</code>.
225 //
226 // - Quando o nome de uma rotina surge numa expressão (que não como operando do
227 // operador de invocação), o compilador interpreta-o como sendo um ponteiro
228 // para essa rotina. Ou seja, salva a excepção já referida, e se estiver
229 // declarada uma rotina com nome `rotina`, então `rotina` e `&rotina` são
230 // expressões equivalentes.
231 //
232 // - Podemos invocar uma rotina através de um seu ponteiro, e.g.,
233 // <code>_ponteiro_para_rotina_</code>, de duas formas totalmente
234 // equivalentes: (a) explicitando o carácter de ponteiro do ponteiro,
235 // desreferenciando-o explicitamente antes de proceder à invocação, i.e.,
236 // <code>(*_ponteiro_para_rotina_)(_lista_de_argumentos_)</code>, ou (b)
237 // procedendo à invocação como se se tratasse de uma rotina, e não de um
238 // ponteiro para ela, i.e.,
239 // <code>_ponteiro_para_rotina_(_lista_de_argumentos_)</code>.
240 //
241 // Devemos usar estas regras sempre com o objectivo de tornar o código mais
242 // claro, pelo que tendemos a preferir as versões mais sintéticas e que menos
243 // explicitam a natureza de ponteiro dos ponteiros para rotinas.
244 //
245 // Magia, mesmo?
246 // -------------
247 //
248 // O pré-processador é mesmo mágico, não é? Pois. É. Demasiado. Evite recorrer a
249 // ele. Use-o apenas onde indispensável (e.g., inclusões de ficheiros de
250 // cabeçalho, evitando as consequências da inclusão múltipla, etc.). O exemplo
251 // de compilação condicional que aqui apresentamos encaixa-se perfeitamente no
252 // conjunto de práticas a evitar sempre que possível. Usámo-lo apenas para
253 // podermos manter num único ficheiro duas versões equivalentes, embora
254 // diferentes, do mesmo código, pois isso traz evidentes vantagens pedagógicas.
255 // Cremos que essas vantagens são suficientes para justificar a violação de uma
256 // boa prática.
