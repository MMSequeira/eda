7 // Inclusão de ficheiros de cabeçalho
8 // ----------------------------------
9 //
10 // Incluímos quatro dos ficheiros de cabeçalho da biblioteca padrão do C, i.e.,
11 // da biblioteca que acompanha a linguagem C tal como definida pela sua norma:
12 //
13 // - `stdio.h` &ndash; Declara os procedimentos `printf()` e `scanf()`, usados
14 // respectivamente para escrever no ecrã (ou melhor, no canal de saída) e
15 // para ler do teclado (ou melhor, do canal de entrada). Declara também
16 // `feof()`, `ferror()`, `fprintf()` e `stderr`.
17 //
18 // - `stdlib.h` &ndash; Define a _macro_ `EXIT_SUCCESS`, que é usada para
19 // assinalar sucesso na execução de um programa, quando este termina, ou
20 // seja, quando se _sai_ do programa, e a macro `EXIT_FAILURE`, que é usada da
21 // mesma forma, mas para assinalar insucesso.
22 //
23 // - `stdbool.h` &ndash; Para podermos usar o valor booleano `true` como
24 // _guarda_ do ciclo `while`.
25 //
26 // - `assert.h` &ndash; Para podermos usar a rotina `assert()` para verificar a
27 // validade dos argumentos passados à função `factorial()`.
28 //
29 #include <stdio.h>
30 #include <stdlib.h>
31 #include <stdbool.h>
32 #include <assert.h>
33 
34 // Definição da função `factorial()`
35 // ---------------------------------
36 //
37 int factorial(const int n)
38 {
39  // Verificamos a validade do argumento recebido usando a rotina especial
40  // (na realidade uma _macro_ do pré-processador) chamada `assert()`. O
41  // limite superior para o valor do argumento é muito baixo: 12. Isso
42  // deve-se a dois factos:
43  //
44  // 1. Os `int` em C, nos nossos ambientes típicos, são guardados em
45  // apenas 32 bit interpretados em complemento para dois, pelo que um
46  // inteiro só pode tomar valores entre ![-2^31](http://bit.ly/16CVI6B) e
47  // ![2^31-1](http://bit.ly/12AavZ0), ou seja, entre
48  // -2&thinsp;147&thinsp;483&thinsp;648 e
49  // 2&thinsp;147&thinsp;483&thinsp;647.
50  //
51  // 2. O factorial é uma função de crescimento muito rápido, pelo que
52  // 13!, com o valor 6&thinsp;227&thinsp;020&thinsp;800, excede já o
53  // limite superior dos `int`.
54  assert(0 <= n && n <= 12);
55 
56  // Começamos por verificar se se aplica o caso especial em que o valor
57  // de `n` é zero.
58  if (n == 0)
59  // Se se aplicar, retornamos imediatamente (i.e., terminamos a função,
60  // retornando ao ponto de invocação) _devolvendo_ o valor 1.
61  return 1;
62 
63  // No caso contrário, retornamos devolvendo o produto entre o valor de
64  // `n` e o valor devolvido por uma nova invocação da mesma função,
65  // embora com um valor mais pequeno passado como argumento, ou seja, _n
66  // - 1_ em vez de _n_. Trata-se de uma invocação recursiva da própria
67  // função.
68  return n * factorial(n - 1);
69 }

71 // A rotina principal `main()`
72 // ---------------------------
73 //
74 int main(void)
75 {
76  // Começamos por escrever no ecrã o nome e o propósito do programa.
77  printf("Hello world! Let's calculate the factorial of a number.\n");
78 
79  // Uma vez que o objectivo do programa é calcular o factorial de um
80  // número, usamos este ciclo para obter do utilizado um número inteiro
81  // não negativo, que ficará guardado na variável `number`.
82  int number;
83  // Usamos um ciclo «infinito», pois o local apropriado para a
84  // verificação da guarda não é nem no início do passo, nem no seu final,
85  // mas sim a meio. (Na realidade o ciclo tem duas saídas: uma se a
86  // leitura tiver sucesso, outra, terminando o programa, se durante a
87  // leitura ocorrer um erro irrecuperável.)
88  while(true) {
89  // Pedimos ao utilizador para introduzir um número inteiro não
90  // negativo. Neste caso não terminamos a linha, pois pretendemos
91  // que a resposta do utilizador surja visualmente à frente do
92  // texto que escrevemos. Em vez disso, escrevemos um espaço,
93  // para que o número introduzido pelo utilizador não surja
94  // «colado» ao nosso texto. A leitura dos caracteres
95  // introduzidos pelo utilizador só é realizada depois de este
96  // carregar em «enter», o que terá o efeito visual de terminar
97  // esta linha.
98  printf("Please enter a non-negative integer "
99  "between 0 and 12: ");
100 
101  // Tentamos ler a resposta do utilizador com um número inteiro.
102  // Se a leitura tiver sucesso e o número estiver dentro da gama
103  // pretendida, interrompemos o ciclo. A função `scanf()` devolve
104  // o número de atribuições a itens de entrada realizadas com
105  // sucesso. No nosso caso podemos saber se foi lido um valor
106  // inteiro com sucesso verificando se o valor devolvido foi 1.
107  // Para saber se o valor lido é válido, verificamos se se
108  // encontra entre 0 e 12. Caso tudo tenha corrido bem, o ciclo é
109  // interrompido através da instrução `break`.
110  if (scanf("%d", &number) == 1 && 0 <= number && number <= 12)
111  break;
112 
113  // Neste ponto sabemos que algum erro ocorreu. Temos de
114  // verificar se esse erro é irrecuperável, o que acontece se o
115  // canal de leitura `stdin` tiver ficado marcado como tendo sido
116  // atingido o fim do ficheiro (a entrada de um programa pode ser
117  // redireccionada de modo a ser feita a partir de um ficheiro)
118  // ou como tendo ocorrido um outro erro de leitura.
119  if (feof(stdin) || ferror(stdin)) {
120  // Se ocorreu um erro irrecuperável, assinalamo-lo no
121  // canal apropriado para o efeito, `stderr`, e
122  // terminamos o programa devolvendo um código de erro.
123  fprintf(stderr,
124  "Unrecoverable error reading user response. "
125  "Exiting.\n");
126 
127  // Terminamos o programa devolvendo o valor especial
128  // `EXIT_FAILURE` para comunicar este erro à entidade
129  // que executou o programa.
130  return EXIT_FAILURE;
131  }
132 
133  // Neste ponto sabemos que a leitura não teve sucesso, mas que
134  // isso não se deveu a um erro irrecuperável. O erro pode ter
135  // ocorrido por o utilizador não ter introduzido um número
136  // inteiro ou por ter introduzido um valor inteiro com um valor
137  // fora da gama válida. Assim, começamos por alertar o
138  // utilizador acerca do seu erro.
139  printf("Invalid response. Try again.\n");
140 
141  // Depois, limpamos o canal de entrada de tudo o que o
142  // utilizador tiver introduzido até ao final da linha, de modo a
143  // que este possa tentar introduzir um novo valor. Para
144  // descartar uma linha completa de caracteres, invocamos o
145  // procedimento `scanf()` com a cadeia de formação
146  // `%*[^\n]%*1[\n]`. O significado é o seguinte:
147  //
148  // - Primeira especificação de conversão, `%*[^\n]`: Ler _e
149  // descartar_ (daí o caractere `*`) todos os caracteres até o
150  // próximo fim-de-linha, exclusive.
151  //
152  // - Segunda especificação de conversão, `%*1[\n]`: Ler _e
153  // descartar_ (daí o caractere `*`) exactamente um caractere
154  // fim-de-linha. Colocar `\n` na cadeia de formatação em vez
155  // desta especificação de conversão não teria o resultado
156  // pretendido, pois o procedimento `scanf()` interpreta todos
157  // os caracteres «brancos» como uma ordem para ignorar um
158  // número arbitrário de caracteres brancos até ser encontrado
159  // o primeiro caractere não branco, o que levaria a que não
160  // surgisse novo pedido de inserção de um número válido
161  // enquanto o utilizador fosse carregando em «enter».
162  scanf("%*[^\n]%*1[\n]");
163  }
164 
165  // Finalmente, podemos calcular o factorial pretendido e escrever o
166  // resultado no ecrã. Para isso invocamos o procedimento `printf()`
167  // passando-lhe três argumentos.
168  //
169  // O primeiro argumento, `"%d! = %d\n"`, é uma cadeia de caracteres de
170  // formação. Esta cadeia da caracteres é interpretada pelo procedimento
171  // `printf()`, que imprime directamente no ecrã os caracteres da cadeia
172  // de formação, excepto quando encontra _especificações de formatação_
173  // que, tal como no caso do procedimento `scanf()`, são introduzidas
174  // pelo caractere `%`. Neste caso temos duas especificações de
175  // formatação, ambas `%d`.
176  //
177  // Quando o procedimento `printf()` encontra a primeira especificação de
178  // formatação, transforma o valor do segundo argumento, que tem de ser
179  // um `int`, uma vez que após o caractere `%` se encontra o caractere
180  // `d`, numa sequência de caracteres e escreve-os no ecrã. Isso leva a
181  // que o valor da variável `number` seja escrito no ecrã, na notação
182  // decimal usual. Quando o procedimento `printf()` encontra a segunda
183  // especificação de formatação, transforma o valor do terceiro argumento
184  // numa sequência de caracteres e escreve-os no ecrã. Isso leva a que o
185  // valor devolvido pela função `factorial()` quando invocada com o
186  // argumento `number` seja escrito no ecrã, na notação decimal usual.
187  //
188  // Por exemplo, admitindo que o valor de `number` é 5, então esta
189  // instrução escreve no ecrã a seguinte linha:
190  //
191  // ```
192  // 5! = 120
193  // ```
194  printf("%d! = %d\n", number, factorial(number));
195 
196  // Terminada a impressão do factorial calculado, podemos terminar o
197  // programa retornando da execução da rotina `main()`. Uma vez que a
198  // execução teve sucesso, devolvemos o valor especial `EXIT_SUCCESS` ao
199  // retornar.
200  return EXIT_SUCCESS;
201 }
