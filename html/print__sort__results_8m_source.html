1 % Plots and prints graphs with sort time statistics. For example:
2 %

8  sorted_results = dlmread('sorted_results.csv', ';');
9 
10  sizes = shuffled_results(:, 1);
11

16  n_types = columns(types);
17 
18  algs = {'bubble', 'selection', 'insertion', 'shell', 'quick', 'merge'};
19  algs_long = {'bubble sort', 'selection sort', 'insertion sort', 'Shell sort', 'quicksort', 'merge sort'};
20  algs_marks = {'.', 'o', 'x', '*', '+', '^'};
21  n_algs = columns(algs);
22 
23  stats = {'comparisons', 'swaps', 'copies', 'runs', 'repetitions', 'average', 'stddev', 'median', 'minimum', 'maximum'};
24  n_stats = columns(stats);
25 
26  types_index = ismember(types, types_to_plot);
27  n_types_to_plot = sum(types_index);
28 
29  algs_index = ismember(algs, algs_to_plot);
30  n_algs_to_plot = sum(algs_index);
31 
32  stats_index = ismember(stats, stats_to_plot);
33  n_stats_to_plot = sum(stats_index);
34 
35  styles = cell(1, n_types * n_algs * n_stats);
36  headers = cell(1, n_types * n_algs * n_stats);
37  for t = 1 : n_types,
38  for a = 1 : n_algs,
39  for s = 1 : n_stats,
40  header = '';
41  if n_algs_to_plot != 1,
42  header = [header algs_long{1, a}];
43  endif
44  if n_types_to_plot != 1,
45  if !isempty(header) header = [header '/']; endif
46  header = [header types{1, t} ' files'];
47  endif
48  if n_stats_to_plot != 1,
49  if !isempty(header) header = [header '/']; endif
50  header = [header stats{1, s}];
51  endif
52  i = (t - 1) * n_algs * n_stats + (a - 1) * n_stats + s;
53  headers{1, i} = header;
54  styles{1, i} = [algs_marks{a} types_colors{t}];
55  endfor
56  endfor
57  endfor
58 
59  if n_stats_to_plot != 1,
60  plot_title = 'Sort time';
61  else
62  plot_title = ['Sort ' stats_to_plot{1} ' time'];
63  endif
64 
65  if n_algs_to_plot == 1,
66  plot_title = [plot_title ' with ' algs_long{algs_index}];
67  endif
68 
69  if n_types_to_plot == 1,
70  plot_title = [plot_title ' for ' types{types_index} ' files'];
71  endif
72 
73  index = logical(kron(types_index, algs_index, stats_index));
74 
75  figure('Position',[0, 0, 1800, 900]);
76 
77  loglog(sizes, results(:, index), styles(:, index));
78 
79  legend(headers(:, index), 'location', 'northeastoutside');
