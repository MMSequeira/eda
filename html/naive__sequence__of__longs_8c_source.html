26 // ### Definição da estrutura `struct naive_sequence_of_long`
27 //
28 // Esta estrutura contém dois campos ou atributos:
29 //
30 // - `terms` &ndash; Um ponteiro para o _array_ dinâmico que contém os termos da
31 // sucessão.
32 //
33 // - `length` &ndash; Inteiro guardando o comprimento actual da sucessão, i.e.,
34 // o seu número de termos. Note que o comprimento da sucessão é sempre
35 // igual ao comprimento do `array` que guarda os termos.
36 struct naive_sequence_of_longs {
37  long *terms;
38  int length;
39 };

60 // ### Implementação do construtor do TAD
61 //
62 struct naive_sequence_of_longs *NSEQL_new(void)
63 {
64  // Reservamos primeiro espaço para uma nova `struct sequence_of_longs`
65  // usando a rotina `malloc()`. Obtemos a quantidade de unidades de
66  // memória a reservar usando o operador `sizeof`. Guardamos o endereço
67  // dessa nova variável dinâmica no ponteiro `sl`, através do qual se
68  // passará a manipular a nova sucessão de `long`. No final do construtor
69  // este ponteiro será devolvido.
70  struct naive_sequence_of_longs *sl =
71  malloc(sizeof(struct naive_sequence_of_longs));
72 
73  // De seguida inicializamos os campos ou atributos da rotina com valores
74  // apropriados para uma sucessão vazia. O comprimento de uma sucessão
75  // vazia é naturalmente 0. Uma vez que não é possível ter um ponteiro
76  // válido para um _array_ dinâmico com 0 itens, inicializa-se o ponteiro
77  // para o primeiro item desse _array_ com o valor `NULL`.
78  sl->length = 0;
79  sl->terms = NULL;
80 
81  // Terminada a construção da nova sucessão, há que devolver o seu
82  // endereço (ou ponteiro), guardado na variável `sl`.
83  return sl;
84 }

93 // ### Implementação do _modificador_ de adição de um novo termo à sucessão
94 //
95 void NSEQL_add(struct naive_sequence_of_longs *sl, long new_term)
96 {
97  // Uma vez que o _array_ dinâmico tem sempre a sua capacidade esgotada, adicionar um novo termo à sucessão implica sempre redimensionar esse _array_. Para aumentar a capacidade, usamos a rotina `realloc()` para reservar um
98  // novo _array_ dinâmico com capacidade para mais um termo _e incluindo
99  // todos os termos que já constam na sucessão_. É este último
100  // requisito que nos leva a usar a rotina `realloc()`, e não a
101  // rotina `malloc()`. A rotina `realloc()`, se precisar de
102  // reservar nova memória, i.e., se não conseguir simplesmente
103  // estender o _array_ existente, copiará automaticamente os
104  // termos da memória original. É por isso que o primeiro
105  // argumento passado a `realloc()` é um ponteiro para o _array_
106  // original.
107  sl->terms = realloc(sl->terms, (sl->length + 1) * sizeof(long));
108 
109  // Finalmente, guardamos o novo termo `new_term` no local apropriado do
110  // _array_, que neste ponto já tem certamente (hmmmm... será?)
111  // capacidade suficiente, i.e., na posição `sl->length`. Depois,
112  // incrementamos o comprimento da sucessão.
113  sl->terms[sl->length++] = new_term;
114 }
